14  package eu.fbk.rdfpro;
15  
16  import javax.annotation.Nullable;
17  
18  import org.openrdf.model.Statement;
19  import org.openrdf.model.Value;
20  import org.openrdf.rio.RDFHandler;
21  import org.openrdf.rio.RDFHandlerException;
22  
23  /**
24   * Reduce function in a MapReduce job.
25   * <p>
26   * A {@code Reducer} object is used in a MapReduce job (see
27   * {@link RDFProcessors#mapReduce(Mapper, Reducer, boolean)}) to process a partition of statements
28   * associated to a certain {@code Value} key produced by a {@link Mapper} in a previous map phase
29   * (e.g., all the statements having a specific subject).
30   * </p>
31   * <p>
32   * Implementations of this interface should be thread-safe, as multiple reduce jobs can be fired
33   * in parallel with method {@code reduce()} being invoked concurrently by different threads on
34   * different statement partitions.
35   * </p>
36   */
37  @FunctionalInterface
38  public interface Reducer {
39  
40      /**
41       * Processes the statement partition associated to a certain key, emitting output statements
42       * to the supplied {@code RDFHandler}.
43       *
44       * @param key
45       *            the partition key, possibly null
46       * @param statements
47       *            a modifiable array with the statements belonging to the partition, not null
48       * @param handler
49       *            the {@code RDFHandler} where to emit output statements, not null
50       * @throws RDFHandlerException
51       *             on error
52       */
53      void reduce(@Nullable Value key, Statement[] statements, RDFHandler handler)
54              throws RDFHandlerException;
55  
56  }
