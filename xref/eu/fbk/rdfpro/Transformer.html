41  /**
42   * A generic {@code Statement -> Statement [0..*]} transformer.
43   * <p>
44   * A {@code Transformer} is a stateless function mapping an input {@code Statement} to zero or
45   * more output {@code Statement}s, via method {@link #transform(Statement, RDFHandler)}; for
46   * efficiency reasons, output statements are not returned as a list but are instead forwarded to a
47   * supplied {@link RDFHandler} (this avoids creation of intermediate array/lists objects, as well
48   * as the efficient mapping of a statement to a large number of output statements). While somehow
49   * similar to an {@link RDFProcessor}, a {@code Transformer} operates on a statement at a time,
50   * whereas an {@code RDFProcessor} transforms streams of statements, i.e., it can perform
51   * transformations that depends on combinations or sets of statements rather than on a single
52   * input statement.
53   * </p>
54   * <p>
55   * Implementations of this interface should be thread-safe, i.e., they should support multiple
56   * threads calling concurrently method {@code transform()}; this can be however achieved by means
57   * of synchronization.
58   * </p>
59   * <p>
60   * A number of constants and static factory methods provide access to basic {@code Transformer}
61   * implementations:
62   * </p>
63   * <ul>
64   * <li>{@link #NIL} is the null transformer, mapping every statement to an empty statement list;</li>
65   * <li>{@link #IDENTITY} is the identity transformer, mapping every statement to itself;</li>
66   * <li>{@link #filter(Predicate)} and {@link #filter(String, Predicate)} create transformers that
67   * apply a given {@code Predicate} to check the input statements or its components, emitting the
68   * statement only if the check is successful;</li>
69   * <li>{@link #map(Function)} and {@link #map(String, Function)} create transformers that apply a
70   * given {@code Function} to transform the input statement or its components, emitting the result
71   * to the supplied {@code RDFHandler};</li>
72   * <li>{@link #set(String, Value...)} create transformers that replace selected components of the
73   * input statement;</li>
74   * <li>{@link #sequence(Transformer...)} chains multiple transformers so that the output of one is
75   * used as the input of the next;</li>
76   * <li>{@link #parallel(Transformer...)} calls multiple transformers in parallel, emitting the
77   * concatenation of their results;</li>
78   * <li>{@link #rules(String)} return a transformer that applies the filtering rules encoded in a
79   * supplied string.</li>
80   * </ul>
81   */
82  @FunctionalInterface
83  public interface Transformer {

90          }

106      * Returns a {@code Transformer} that emits only statements matching the supplied

113     static Transformer filter(final Predicate<? super Statement> predicate) {
114         Objects.requireNonNull(predicate);
115         return new Transformer() {
116

118             public void transform(final Statement statement, final RDFHandler handler)

121                     handler.handleStatement(statement);
122                 }

128     /**
129      * Returns a {@code Transformer} that emits only statements whose selected component matches
130      * the specified {@code Predicate}. The predicate is evaluated on each {@code s}, {@code p},
131      * {@code o} , {@code c} component given by the {@code components} string. If any of these
132      * tests fails, the input statement is dropped, otherwise it is emitted to the
133      * {@code RDFHandler}.
134      *
135      * @param components
136      *            a string of symbols {@code s} , {@code p}, {@code o}, {@code c} specifying which
137      *            components to test, not null
138      * @param predicate
139      *            the predicate to apply to selected components, not null
140      * @return the created {@code Transformer}
141      */
142     static Transformer filter(final String components, final Predicate<? super Value> predicate) {
143 
144         Objects.requireNonNull(predicate);
145 
146         final String comp = components.trim().toLowerCase();
147         final boolean[] flags = new boolean[4];
148         for (int i = 0; i < comp.length(); ++i) {
149             final char c = comp.charAt(i);
150             final int index = c == 's' ? 0 : c == 'p' ? 1 : c == 'o' ? 2
151                     : c == 'c' || c == 'g' ? 3 : -1;
152             if (index < 0 || flags[index]) {
153                 throw new IllegalArgumentException("Invalid components '" + components + "'");
154             }
155             flags[index] = true;
156         }
157 
158         if (!flags[0] && !flags[1] && !flags[2] && !flags[3]) {
159             return IDENTITY;
160         }
161 
162         return new Transformer() {
163 
164             private final boolean skipSubj = !flags[0];
165 
166             private final boolean skipPred = !flags[1];
167 
168             private final boolean skipObj = !flags[2];
169 
170             private final boolean skipCtx = !flags[3];
171 
172             @Override
173             public void transform(final Statement statement, final RDFHandler handler)
174                     throws RDFHandlerException {
175                 if ((skipSubj || predicate.test(statement.getSubject()))
176                         && (skipPred || predicate.test(statement.getPredicate()))
177                         && (skipObj || predicate.test(statement.getObject()))
178                         && (skipCtx || predicate.test(statement.getContext()))) {
179                     handler.handleStatement(statement);
180                 }
181             }
182 
183         };
184     }

200             @Override
201             public void transform(final Statement statement, final RDFHandler handler)
202                     throws RDFHandlerException {

207             }

212     /**
213      * Returns a {@code Transformer} that replaces selected components of the input statement
214      * using the supplied function. The function is applied to each {@code s}, {@code p},
215      * {@code o} , {@code c} component given by the {@code components} string. The resulting
216      * component values, if not null and compatible with expected component types, are used to
217      * assemble an output statement that is emitted to the supplied {@code RDFHandler}.
218      *
219      * @param components
220      *            a string of symbols {@code s} , {@code p}, {@code o}, {@code c} specifying the
221      *            components to replace, not null
222      * @param function
223      *            the function to apply to selected components, not null
224      * @return the created {@code Transformer}
225      */
226     static Transformer map(final String components,
227             final Function<? super Value, ? extends Value> function) {
228 
229         Objects.requireNonNull(function);
230 
231         final String comp = components.trim().toLowerCase();
232         final boolean[] flags = new boolean[4];
233         for (int i = 0; i < comp.length(); ++i) {
234             final char c = comp.charAt(i);
235             final int index = c == 's' ? 0 : c == 'p' ? 1 : c == 'o' ? 2
236                     : c == 'c' || c == 'g' ? 3 : -1;
237             if (index < 0 || flags[index]) {
238                 throw new IllegalArgumentException("Invalid components '" + components + "'");
239             }
240             flags[index] = true;
241         }
242 
243         if (!flags[0] && !flags[1] && !flags[2] && !flags[3]) {
244             return IDENTITY;
245         }
246 
247         return new Transformer() {
248 
249             private final boolean mapSubj = flags[0];
250 
251             private final boolean mapPred = flags[1];
252 
253             private final boolean mapObj = flags[2];
254 
255             private final boolean mapCtx = flags[3];
256 
257             @Override
258             public void transform(final Statement statement, final RDFHandler handler)
259                     throws RDFHandlerException {
260 
261                 Resource subj = statement.getSubject();
262                 URI pred = statement.getPredicate();
263                 Value obj = statement.getObject();
264                 Resource ctx = statement.getContext();
265 
266                 boolean modified = false;
267 
268                 if (this.mapSubj) {
269                     final Value v = function.apply(subj);
270                     if (!(v instanceof Resource)) {
271                         return;
272                     }
273                     modified |= v != subj;
274                     subj = (Resource) v;
275                 }
276 
277                 if (this.mapPred) {
278                     final Value v = function.apply(pred);
279                     if (!(v instanceof URI)) {
280                         return;
281                     }
282                     modified |= v != pred;
283                     pred = (URI) v;
284                 }
285 
286                 if (this.mapObj) {
287                     final Value v = function.apply(obj);
288                     if (v == null) {
289                         return;
290                     }
291                     modified |= v != obj;
292                     obj = v;
293                 }
294 
295                 if (this.mapCtx) {
296                     final Value v = function.apply(ctx);
297                     if (!(v instanceof Resource)) {
298                         return;
299                     }
300                     modified |= v != ctx;
301                     ctx = (Resource) v;
302                 }
303 
304                 if (!modified) {
305                     handler.handleStatement(statement);
306                 } else if (ctx == null) {
307                     handler.handleStatement(Statements.VALUE_FACTORY.createStatement(subj, pred,
308                             obj));
309                 } else {
310                     handler.handleStatement(Statements.VALUE_FACTORY.createStatement(subj, pred,
311                             obj, ctx));
312                 }
313             }
314 
315         };
316     }

318     /**
319      * Returns a {@code Transformer} that replaces selected components of the input statement with
320      * the replacement values supplied. Parameter {@code components} is a sequence of {@code s},
321      * {@code p}, {@code o}, {@code c} symbols that specify the components to replace. The i-th
322      * symbol corresponds to a replacement i-th value in the array {@code values} (if the array is
323      * shorter, its last element is used). An output statement is assembled by mixing unmodified
324      * components with replaced ones. If a valid statement is obtained, it is emitted to the
325      * supplied {@code RDFHandler}.
326      *
327      * @param components
328      *            a string of symbols {@code s} , {@code p}, {@code o}, {@code c} specifying the
329      *            components to replace, not null
330      * @param values
331      *            the replacement values, with i-th value corresponding to the i-th symbol (if
332      *            fewer values are supplied, the last value is used for unmatched component
333      *            symbols).
334      * @return the created {@code Transformer}
335      */
336     static Transformer set(final String components, final Value... values) {
337 
338         final boolean[] flags = new boolean[4];
339         final Value[] vals = new Value[4];
340 
341         final String comp = components.trim().toLowerCase();
342         for (int i = 0; i < comp.length(); ++i) {
343             final char c = comp.charAt(i);
344             final int index = c == 's' ? 0 : c == 'p' ? 1 : c == 'o' ? 2
345                     : c == 'c' || c == 'g' ? 3 : -1;
346             if (index < 0 || flags[index]) {
347                 throw new IllegalArgumentException("Invalid components '" + components + "'");
348             }
349             flags[index] = true;
350             vals[index] = values[Math.min(values.length - 1, i)];
351         }
352 
353         if (!flags[0] && !flags[1] && !flags[2] && !flags[3]) {
354             return IDENTITY;
355 
356         } else if (vals[0] != null && !(vals[0] instanceof Resource) || vals[1] != null
357                 && !(vals[1] instanceof URI) || vals[3] != null && !(vals[3] instanceof Resource)) {
358             return NIL;
359 
360         } else {
361             return new Transformer() {
362 
363                 private final boolean subjFlag = flags[0];
364 
365                 private final boolean predFlag = flags[1];
366 
367                 private final boolean objFlag = flags[2];
368 
369                 private final boolean ctxFlag = flags[3];
370 
371                 private final Resource subjVal = (Resource) vals[0];
372 
373                 private final URI predVal = (URI) vals[1];
374 
375                 private final Value objVal = vals[2];
376 
377                 private final Resource ctxVal = (Resource) vals[3];
378 
379                 @Override
380                 public void transform(final Statement statement, final RDFHandler handler)
381                         throws RDFHandlerException {
382 
383                     final Resource subj = this.subjFlag ? this.subjVal : statement.getSubject();
384                     final URI pred = this.predFlag ? this.predVal : statement.getPredicate();
385                     final Value obj = this.objFlag ? this.objVal : statement.getObject();
386                     final Resource ctx = this.ctxFlag ? this.ctxVal : statement.getContext();
387 
388                     final ValueFactory vf = Statements.VALUE_FACTORY;
389                     if (ctx == null) {
390                         handler.handleStatement(vf.createStatement(subj, pred, obj));
391                     } else {
392                         handler.handleStatement(vf.createStatement(subj, pred, obj, ctx));
393                     }
394                 }
395 
396             };
397         }
398     }

400     /**
401      * Returns a {@code Transformer} chaining the specified {@code Transformer}s. The first
402      * supplied transformer is applied first, its output piped to the next transformer and so on,
403      * with the final result sent to the supplied {@code RDFHandler}. If no transformer is
404      * supplied, {@link #NIL} is returned.
405      *
406      * @param transformers
407      *            the transformers to chain
408      * @return the created {@code Transformer}
409      */
410     static Transformer sequence(final Transformer... transformers) {
411         if (Arrays.asList(transformers).contains(null)) {
412             throw new NullPointerException();
413         }
414         if (transformers.length == 0) {
415             return NIL;
416         } else if (transformers.length == 1) {
417             return transformers[0];
418         } else if (transformers.length == 2) {
419             final Transformer first = transformers[0];
420             final Transformer second = transformers[1];
421             return new Transformer() {
422 
423                 @Override
424                 public void transform(final Statement statement, final RDFHandler handler)
425                         throws RDFHandlerException {
426                     final List<Statement> buffer = new ArrayList<>();
427                     final RDFHandler collector = RDFHandlers.wrap(buffer);
428                     first.transform(statement, collector);
429                     for (final Statement intermediateStatement : buffer) {
430                         second.transform(intermediateStatement, handler);
431                     }
432                 }
433 
434             };
435         } else {
436             final Transformer[] newTransformers = new Transformer[transformers.length - 1];
437             System.arraycopy(transformers, 2, newTransformers, 1, transformers.length - 2);
438             newTransformers[0] = sequence(transformers[0], transformers[1]);
439             return sequence(newTransformers); // recursion
440         }
441     }

443     /**
444      * Returns a {@code Transfomer} that applies each supplied transformer in parallel, emitting
445      * the concatenation of their results to the supplied {@code RDFHandler}.
446      *
447      * @param transformers
448      *            the transformers to compose in parallel
449      * @return the created {@code Transformer}
450      */
451     static Transformer parallel(final Transformer... transformers) {
452         if (Arrays.asList(transformers).contains(null)) {
453             throw new NullPointerException();
454         }
455         if (transformers.length == 0) {
456             return NIL;
457         } else if (transformers.length == 1) {
458             return transformers[0];
459         } else {
460             return new Transformer() {
461 
462                 @Override
463                 public void transform(final Statement statement, final RDFHandler handler)
464                         throws RDFHandlerException {
465                     for (final Transformer transformer : transformers) {
466                         transformer.transform(statement, handler);
467                     }
468                 }
469 
470             };
471         }
472     }

474     /**
475      * Returns a {@code Transformer} applying the filtering and assignent rules encoded by the
476      * supplied string. Given {@code X} a quad component (values: {@code s}, {@code p}, {@code o},
477      * {@code c}), the string contains three types of rules:
478      * <ul>
479      * <li>{@code +X value list} - quad is dropped if {@code X} does not belong to {@code list};</li>
480      * <li>{@code -X value list} - quad is dropped if {@code X} belongs to {@code list};</li>
481      * <li>{@code =X value} - quad component {@code X} is replaced with {@code value} (evaluated
482      * after filters).</li>
483      * </ul>
484      * Values must be encoded in Turtle. The following wildcard values are supported:
485      * <ul>
486      * <li>{@code <*>} - any URI;</li>
487      * <li>{@code _:*} - any BNode;</li>
488      * <li>{@code *} - any literal;</li>
489      * <li>{@code *@*} - any literal with a language;</li>
490      * <li>{@code *@xyz} - any literal with language {@code xyz};</li>
491      * <li>{@code *^^*} - any typed literal;</li>
492      * <li>{@code *^^<uri>} - any literal with datatype {@code <uri>};</li>
493      * <li>{@code *^^ns:uri} - any literal with datatype {@code ns:uri};</li>
494      * <li>{@code *^^ns:*} - any typed literal with datatype prefixed with {@code ns:};</li>
495      * <li>{@code ns:*} - any URI prefixed with {@code ns:};</li>
496      * <li>{@code <ns*>} - any URI with namespace URI {@code ns}.</li>
497      * </ul>
498      *
499      * @param rules
500      *            the filtering rules, not null
501      * @return the created {@code Transformer}
502      */
503     static Transformer rules(final String rules) {
504         return new RuleTransformer(rules);
505     }
506 
507     /**
508      * Applies the {@code Transformer} to the specified statement, emitting its output (possibly
509      * empty) to the supplied {@code RDFHandler}. Statement emission should be performed calling
510      * method {@link RDFHandler#handleStatement(Statement)}; other {@code RDFHandler} methods
511      * should not be called. This method is meant to be called concurrently by multiple thread, so
512      * a thread-safe implementation is required.
513      *
514      * @param statement
515      *            the input statement, not null
516      * @param handler
517      *            the {@code RDFHandler} where to emit output statements
518      * @throws RDFHandlerException
519      *             in case of error
520      */
521     void transform(Statement statement, RDFHandler handler) throws RDFHandlerException;
522 
523 }

525 final class RuleTransformer implements Transformer {
526 
527     @Nullable
528     private final ValueTransformer subjectTransformer;
529 
530     @Nullable
531     private final ValueTransformer predicateTransformer;
532 
533     @Nullable
534     private final ValueTransformer objectTransformer;
535 
536     @Nullable
537     private final ValueTransformer contextTransformer;
538 
539     @SuppressWarnings("unchecked")
540     public RuleTransformer(final String spec) {
541 
542         // Initialize the arrays used to create the ValueTransformers
543         final List<?>[] expressions = new List<?>[4];
544         final Value[] replacements = new Value[4];
545         final Boolean[] includes = new Boolean[4];
546         for (int i = 0; i < 4; ++i) {
547             expressions[i] = new ArrayList<String>();
548         }
549 
550         // Parse the specification string
551         char action = 0;
552         final List<Integer> components = new ArrayList<Integer>();
553         for (final String token : spec.split("\\s+")) {
554             final char ch0 = token.charAt(0);
555             if (ch0 == '+' || ch0 == '-' || ch0 == '=') {
556                 action = ch0;
557                 if (token.length() == 1) {
558                     throw new IllegalArgumentException("No component(s) specified in '" + spec
559                             + "'");
560                 }
561                 components.clear();
562                 for (int i = 1; i < token.length(); ++i) {
563                     final char ch1 = Character.toLowerCase(token.charAt(i));
564                     final int component = ch1 == 's' ? 0 : ch1 == 'p' ? 1 : ch1 == 'o' ? 2
565                             : ch1 == 'c' ? 3 : -1;
566                     if (component < 0) {
567                         throw new IllegalArgumentException("Invalid component '" + ch1 + "' in '"
568                                 + spec + "'");
569                     }
570                     components.add(component);
571                 }
572             } else if (action == 0) {
573                 throw new IllegalArgumentException("Missing selector in '" + spec + "'");
574             } else if (action == '=') {
575                 for (final int component : components) {
576                     replacements[component] = Statements.parseValue(token, Namespaces.DEFAULT);
577                 }
578             } else {
579                 for (final int component : components) {
580                     ((List<String>) expressions[component]).add(token);
581                     final Boolean include = action == '+' ? Boolean.TRUE : Boolean.FALSE;
582                     if (includes[component] != null
583                             && !Objects.equals(includes[component], include)) {
584                         throw new IllegalArgumentException(
585                                 "Include (+) and exclude (-) rules both "
586                                         + "specified for same component in '" + spec + "'");
587                     }
588                     includes[component] = include;
589                 }
590             }
591         }
592 
593         // Create ValueTransformers
594         final ValueTransformer[] transformers = new ValueTransformer[4];
595         for (int i = 0; i < 4; ++i) {
596             transformers[i] = expressions[i].isEmpty() && replacements[i] == null ? null
597                     : new ValueTransformer((List<String>) expressions[i], replacements[i],
598                             Boolean.TRUE.equals(includes[i]));
599         }
600         this.subjectTransformer = transformers[0];
601         this.predicateTransformer = transformers[1];
602         this.objectTransformer = transformers[2];
603         this.contextTransformer = transformers[3];
604     }
605 
606     @Override
607     public void transform(final Statement statement, final RDFHandler handler)
608             throws RDFHandlerException {
609 
610         // Transform the subject; abort if result is null
611         final Resource oldSubj = statement.getSubject();
612         Resource newSubj = oldSubj;
613         if (this.subjectTransformer != null) {
614             newSubj = (Resource) this.subjectTransformer.transform(oldSubj);
615             if (newSubj == null) {
616                 return;
617             }
618         }
619 
620         // Transform the predicate; abort if result is null
621         final URI oldPred = statement.getPredicate();
622         URI newPred = oldPred;
623         if (this.predicateTransformer != null) {
624             newPred = (URI) this.predicateTransformer.transform(oldPred);
625             if (newPred == null) {
626                 return;
627             }
628         }
629 
630         // Transform the object; abort if result is null
631         final Value oldObj = statement.getObject();
632         Value newObj = oldObj;
633         if (this.objectTransformer != null) {
634             newObj = this.objectTransformer.transform(oldObj);
635             if (newObj == null) {
636                 return;
637             }
638         }
639 
640         // Transform the context; if null use SESAME.NIL; abort if result is null
641         Resource oldCtx = statement.getContext();
642         oldCtx = oldCtx != null ? oldCtx : SESAME.NIL;
643         Resource newCtx = oldCtx;
644         if (this.contextTransformer != null) {
645             newCtx = (Resource) this.contextTransformer.transform(oldCtx);
646             if (newCtx == null) {
647                 return;
648             }
649         }
650 
651         // Return the possibly modified statement
652         if (newSubj == oldSubj && newPred == oldPred && newObj == oldObj && newCtx == oldCtx) {
653             handler.handleStatement(statement); // unchanged;
654         } else if (newCtx.equals(SESAME.NIL)) {
655             handler.handleStatement(Statements.VALUE_FACTORY.createStatement(newSubj, newPred,
656                     newObj));
657         } else {
658             handler.handleStatement(Statements.VALUE_FACTORY.createStatement(newSubj, newPred,
659                     newObj, newCtx));
660         }
661     }

663     private static class ValueTransformer {
664 
665         @Nullable
666         private final Value replacement;
667 
668         private final boolean include;
669 
670         // for URIs
671 
672         private final boolean matchAnyURI;
673 
674         private final Set<String> matchedURINamespaces;
675 
676         private final Set<URI> matchedURIs;
677 
678         // for BNodes
679 
680         private final boolean matchAnyBNode;
681 
682         private final Set<BNode> matchedBNodes;
683 
684         // for Literals
685 
686         private final boolean matchAnyPlainLiteral;
687 
688         private final boolean matchAnyLangLiteral;
689 
690         private final boolean matchAnyTypedLiteral;
691 
692         private final Set<String> matchedLanguages;
693 
694         private final Set<URI> matchedDatatypeURIs;
695 
696         private final Set<String> matchedDatatypeNamespaces;
697 
698         private final Set<Literal> matchedLiterals;
699 
700         ValueTransformer(final Iterable<String> matchExpressions,
701                 @Nullable final Value replacement, final boolean include) {
702 
703             this.replacement = replacement;
704             this.include = include;
705 
706             this.matchedURINamespaces = new HashSet<>();
707             this.matchedURIs = new HashSet<>();
708             this.matchedBNodes = new HashSet<>();
709             this.matchedLanguages = new HashSet<>();
710             this.matchedDatatypeURIs = new HashSet<>();
711             this.matchedDatatypeNamespaces = new HashSet<>();
712             this.matchedLiterals = new HashSet<>();
713 
714             boolean matchAnyURI = false;
715             boolean matchAnyBNode = false;
716             boolean matchAnyPlainLiteral = false;
717             boolean matchAnyLangLiteral = false;
718             boolean matchAnyTypedLiteral = false;
719 
720             for (final String expression : matchExpressions) {
721                 if ("<*>".equals(expression)) {
722                     matchAnyURI = true;
723                 } else if ("_:*".equals(expression)) {
724                     matchAnyBNode = true;
725                 } else if ("*".equals(expression)) {
726                     matchAnyPlainLiteral = true;
727                 } else if ("*@*".equals(expression)) {
728                     matchAnyLangLiteral = true;
729                 } else if ("*^^*".equals(expression)) {
730                     matchAnyTypedLiteral = true;
731                 } else if (expression.startsWith("*@")) {
732                     this.matchedLanguages.add(expression.substring(2));
733                 } else if (expression.startsWith("*^^")) {
734                     if (expression.endsWith(":*")) {
735                         this.matchedDatatypeNamespaces.add(Namespaces.DEFAULT.uriFor(expression
736                                 .substring(3, expression.length() - 2)));
737                     } else {
738                         this.matchedDatatypeURIs.add((URI) Statements.parseValue(
739                                 expression.substring(3), Namespaces.DEFAULT));
740                     }
741                 } else if (expression.endsWith(":*")) {
742                     this.matchedURINamespaces.add(Namespaces.DEFAULT.uriFor(expression.substring(
743                             0, expression.length() - 2)));
744 
745                 } else if (expression.endsWith("*>")) {
746                     this.matchedURINamespaces
747                             .add(expression.substring(1, expression.length() - 2));
748                 } else {
749                     final Value value = Statements.parseValue(expression, Namespaces.DEFAULT);
750                     if (value instanceof URI) {
751                         this.matchedURIs.add((URI) value);
752                     } else if (value instanceof BNode) {
753                         this.matchedBNodes.add((BNode) value);
754                     } else if (value instanceof Literal) {
755                         this.matchedLiterals.add((Literal) value);
756                     }
757 
758                 }
759             }
760 
761             this.matchAnyURI = matchAnyURI;
762             this.matchAnyBNode = matchAnyBNode;
763             this.matchAnyPlainLiteral = matchAnyPlainLiteral;
764             this.matchAnyLangLiteral = matchAnyLangLiteral;
765             this.matchAnyTypedLiteral = matchAnyTypedLiteral;
766         }
767 
768         @Nullable
769         Value transform(final Value value) {
770             final boolean matched = match(value);
771             return this.include && !matched || !this.include && matched ? null
772                     : this.replacement == null ? value : this.replacement;
773         }
774 
775         private boolean match(final Value value) {
776             if (value instanceof URI) {
777                 return this.matchAnyURI //
778                         || contains(this.matchedURIs, value)
779                         || containsNs(this.matchedURINamespaces, (URI) value);
780             } else if (value instanceof Literal) {
781                 final Literal lit = (Literal) value;
782                 final String lang = lit.getLanguage();
783                 final URI dt = lit.getDatatype();
784                 return lang == null && dt == null
785                         && this.matchAnyPlainLiteral //
786                         || lang != null //
787                         && (this.matchAnyLangLiteral || contains(this.matchedLanguages, lang)) //
788                         || dt != null //
789                         && (this.matchAnyTypedLiteral || contains(this.matchedDatatypeURIs, dt) || containsNs(
790                                 this.matchedDatatypeNamespaces, dt)) //
791                         || contains(this.matchedLiterals, lit);
792             } else {
793                 return this.matchAnyBNode //
794                         || contains(this.matchedBNodes, value);
795             }
796         }
797 
798         private static boolean contains(final Set<?> set, final Object value) {
799             return !set.isEmpty() && set.contains(value);
800         }
801 
802         private static boolean containsNs(final Set<String> set, final URI uri) {
803             if (set.isEmpty()) {
804                 return false;
805             }
806             if (set.contains(uri.getNamespace())) {
807                 return true; // exact lookup
808             }
809             final String uriString = uri.stringValue();
810             for (final String elem : set) {
811                 if (uriString.startsWith(elem)) {
812                     return true; // prefix match
813                 }
814             }
815             return false;
816         }
