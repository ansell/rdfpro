1   package eu.fbk.rdfpro.rules.functions;

3   import java.util.ArrayList;
4   import java.util.HashSet;
5   import java.util.List;
6   import java.util.Set;

8   import org.openrdf.model.URI;
9   import org.openrdf.model.Value;
10  import org.openrdf.model.ValueFactory;

13  import org.openrdf.model.vocabulary.XMLSchema;

19  public class CompatibleDatatypeFunction implements Function {
20  
21      private static final Set<Pair> COMPATIBILITIES;
22  
23      static {
24          final Set<Pair> set = new HashSet<>();
25  
26          set.add(new Pair(RDFS.LITERAL, RDF.XMLLITERAL));
27          set.add(new Pair(RDFS.LITERAL, XMLSchema.DURATION));
28          set.add(new Pair(RDFS.LITERAL, XMLSchema.DATETIME));
29          set.add(new Pair(RDFS.LITERAL, XMLSchema.DATE));
30          set.add(new Pair(RDFS.LITERAL, XMLSchema.TIME));
31          set.add(new Pair(RDFS.LITERAL, XMLSchema.GYEARMONTH));
32          set.add(new Pair(RDFS.LITERAL, XMLSchema.GMONTHDAY));
33          set.add(new Pair(RDFS.LITERAL, XMLSchema.GYEAR));
34          set.add(new Pair(RDFS.LITERAL, XMLSchema.GMONTH));
35          set.add(new Pair(RDFS.LITERAL, XMLSchema.GDAY));
36          set.add(new Pair(RDFS.LITERAL, XMLSchema.BOOLEAN));
37          set.add(new Pair(RDFS.LITERAL, XMLSchema.BASE64BINARY));
38          set.add(new Pair(RDFS.LITERAL, XMLSchema.HEXBINARY));
39          set.add(new Pair(RDFS.LITERAL, XMLSchema.FLOAT));
40          set.add(new Pair(RDFS.LITERAL, XMLSchema.DOUBLE));
41          set.add(new Pair(RDFS.LITERAL, XMLSchema.DECIMAL));
42          set.add(new Pair(RDFS.LITERAL, XMLSchema.ANYURI));
43          set.add(new Pair(RDFS.LITERAL, XMLSchema.QNAME));
44          set.add(new Pair(RDFS.LITERAL, XMLSchema.NOTATION));
45          set.add(new Pair(RDFS.LITERAL, XMLSchema.STRING));
46          set.add(new Pair(XMLSchema.DECIMAL, XMLSchema.INTEGER));
47          set.add(new Pair(XMLSchema.INTEGER, XMLSchema.NON_POSITIVE_INTEGER));
48          set.add(new Pair(XMLSchema.INTEGER, XMLSchema.NON_NEGATIVE_INTEGER));
49          set.add(new Pair(XMLSchema.INTEGER, XMLSchema.LONG));
50          set.add(new Pair(XMLSchema.NON_POSITIVE_INTEGER, XMLSchema.NEGATIVE_INTEGER));
51          set.add(new Pair(XMLSchema.NON_NEGATIVE_INTEGER, XMLSchema.POSITIVE_INTEGER));
52          set.add(new Pair(XMLSchema.NON_NEGATIVE_INTEGER, XMLSchema.UNSIGNED_LONG));
53          set.add(new Pair(XMLSchema.UNSIGNED_LONG, XMLSchema.UNSIGNED_INT));
54          set.add(new Pair(XMLSchema.UNSIGNED_INT, XMLSchema.UNSIGNED_SHORT));
55          set.add(new Pair(XMLSchema.UNSIGNED_SHORT, XMLSchema.UNSIGNED_BYTE));
56          set.add(new Pair(XMLSchema.LONG, XMLSchema.INT));
57          set.add(new Pair(XMLSchema.INT, XMLSchema.SHORT));
58          set.add(new Pair(XMLSchema.SHORT, XMLSchema.BYTE));
59          set.add(new Pair(XMLSchema.STRING, XMLSchema.NORMALIZEDSTRING));
60          set.add(new Pair(XMLSchema.NORMALIZEDSTRING, XMLSchema.TOKEN));
61          set.add(new Pair(XMLSchema.TOKEN, XMLSchema.LANGUAGE));
62          set.add(new Pair(XMLSchema.TOKEN, XMLSchema.NAME));
63          set.add(new Pair(XMLSchema.TOKEN, XMLSchema.NMTOKEN));
64          set.add(new Pair(XMLSchema.NAME, XMLSchema.NCNAME));
65          set.add(new Pair(XMLSchema.NCNAME, XMLSchema.ID));
66          set.add(new Pair(XMLSchema.NCNAME, XMLSchema.IDREF));
67          set.add(new Pair(XMLSchema.NCNAME, XMLSchema.ENTITY));
68  
69          while (true) {
70              final int sizeBefore = set.size();
71              final List<Pair> pairs = new ArrayList<>(set);
72              for (final Pair pair1 : pairs) {
73                  for (final Pair pair2 : pairs) {
74                      if (pair1.child.equals(pair2.parent)) {
75                          set.add(new Pair(pair1.parent, pair2.child));
76                      }
77                  }
78              }
79              if (set.size() == sizeBefore) {
80                  break;
81              }
82          }
83  
84          for (final URI uri : new URI[] { RDFS.LITERAL, RDF.XMLLITERAL, XMLSchema.DURATION,
85                  XMLSchema.DATETIME, XMLSchema.DATE, XMLSchema.TIME, XMLSchema.GYEARMONTH,
86                  XMLSchema.GMONTHDAY, XMLSchema.GYEAR, XMLSchema.GMONTH, XMLSchema.GDAY,
87                  XMLSchema.BOOLEAN, XMLSchema.BASE64BINARY, XMLSchema.HEXBINARY, XMLSchema.FLOAT,
88                  XMLSchema.DOUBLE, XMLSchema.DECIMAL, XMLSchema.ANYURI, XMLSchema.QNAME,
89                  XMLSchema.NOTATION, XMLSchema.STRING, XMLSchema.INTEGER,
90                  XMLSchema.NON_POSITIVE_INTEGER, XMLSchema.NON_NEGATIVE_INTEGER, XMLSchema.LONG,
91                  XMLSchema.NEGATIVE_INTEGER, XMLSchema.POSITIVE_INTEGER, XMLSchema.UNSIGNED_LONG,
92                  XMLSchema.UNSIGNED_INT, XMLSchema.UNSIGNED_SHORT, XMLSchema.UNSIGNED_BYTE,
93                  XMLSchema.INT, XMLSchema.SHORT, XMLSchema.BYTE, XMLSchema.NORMALIZEDSTRING,
94                  XMLSchema.TOKEN, XMLSchema.LANGUAGE, XMLSchema.NAME, XMLSchema.NMTOKEN,
95                  XMLSchema.NCNAME, XMLSchema.ID, XMLSchema.IDREF, XMLSchema.ENTITY, }) {
96              set.add(new Pair(uri, uri));
97          }
98  
99          COMPATIBILITIES = set;
100     }
101 
102     @Override
103     public String getURI() {
104         return RR.COMPATIBLE_DATATYPE.stringValue();
105     }
106 
107     @Override
108     public Value evaluate(final ValueFactory factory, final Value... args)
109             throws ValueExprEvaluationException {
110         final URI d1 = (URI) args[0]; // parent
111         final URI d2 = (URI) args[1]; // child
112         final Pair pair = new Pair(d1, d2);
113         return factory.createLiteral(COMPATIBILITIES.contains(pair));
114     }
115 
116     private static final class Pair {
117 
118         public URI parent;
119 
120         public URI child;
121 
122         public Pair(final URI parent, final URI child) {
123             this.parent = parent;
124             this.child = child;
125         }
126 
127         @Override
128         public boolean equals(final Object object) {
129             if (object == this) {
130                 return true;
131             }
132             if (!(object instanceof Pair)) {
133                 return false;
134             }
135             final Pair other = (Pair) object;
136             return this.parent.equals(other.parent) && this.child.equals(other.child);
137         }
138 
139         @Override
140         public int hashCode() {
141             return this.parent.hashCode() * 37 + this.child.hashCode();
142         }
143 
144     }
