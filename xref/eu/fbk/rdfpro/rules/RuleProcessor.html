53      static RDFProcessor create(final String name, final String... args) throws RDFHandlerException {
54  
55          // Validate and parse options
56          final Options options = Options.parse("r!|B!|p!|g!|t|C|c!|b!|w|*", args);
57  
58          // Read base and preserve BNodes settings
59          final boolean preserveBNodes = !options.hasOption("w");
60          String base = options.getOptionArg("b", String.class);
61          base = base == null ? null : Statements.parseValue(base.contains(":") ? base : base + ":",
62                  Namespaces.DEFAULT).stringValue();
63  
64          // Read bindings
65          final String parameters = options.getOptionArg("B", String.class, "");
66          final MapBindingSet bindings = new MapBindingSet();
67          for (final String token : parameters.split("\\s+")) {
68              final int index = token.indexOf('=');
69              if (index >= 0) {
70                  bindings.addBinding(token.substring(0, index).trim(), Statements.parseValue(token
71                          .substring(index + 1).trim(), Namespaces.DEFAULT));
72              }
73          }
74  
75          // Read rulesets
76          final List<String> rulesetURLs = new ArrayList<>();
77          final String rulesetNames = options.getOptionArg("r", String.class);
78          for (final String rulesetName : rulesetNames.split(",")) {
79              String location = Environment.getProperty("rdfpro.rules." + rulesetName);
80              location = location != null ? location : rulesetName;
81              rulesetURLs.add(IO.extractURL(location).toString());
82          }
83          final RDFSource rulesetSource = RDFSources.read(true, preserveBNodes, base, null,
84                  rulesetURLs.toArray(new String[0]));
85          Ruleset ruleset = Ruleset.fromRDF(rulesetSource);
86  
87          // Transform ruleset
88          ruleset = ruleset.transform(bindings);
89          URI globalURI = null;
90          if (options.hasOption("G")) {
91              final String u = options.getOptionArg("G", String.class);
92              globalURI = (URI) Statements.parseValue(u.contains(":") ? u //
93                      : u + ":", Namespaces.DEFAULT);
94          }
95          final String mode = options.getOptionArg("g", String.class, "none").trim();
96          if ("global".equalsIgnoreCase(mode)) {
97              ruleset = ruleset.transformGlobalGM(globalURI);
98          } else if ("separate".equalsIgnoreCase(mode)) {
99              ruleset = ruleset.transformSeparateGM();
100         } else if ("star".equalsIgnoreCase(mode)) {
101             ruleset = ruleset.transformStarGM(globalURI);
102         } else if (!"none".equalsIgnoreCase(mode)) {
103             throw new IllegalArgumentException("Unknown graph inference mode: " + mode);
104         }
105 
106         // Read static closure settings
107         boolean emitStatic = false;
108         URI staticContext = null;
109         if (options.hasOption("C")) {
110             emitStatic = true;
111         } else if (options.hasOption("c")) {
112             emitStatic = true;
113             final String ctx = options.getOptionArg("c", String.class);
114             staticContext = (URI) Statements.parseValue(ctx.contains(":") ? ctx //
115                     : ctx + ":", Namespaces.DEFAULT);
116         }
117 
118         // Read bnode types settings
119         final boolean dropBNodeTypes = options.hasOption("t");
120 
121         // Read Mapper for optional partitioning
122         Mapper mapper = null;
123         final String partitioning = options.getOptionArg("p", String.class, "none").trim();
124         if ("entity".equalsIgnoreCase(partitioning)) {
125             mapper = Mapper.concat(Mapper.select("s"), Mapper.select("o"));
126         } else if ("graph".equalsIgnoreCase(partitioning)) {
127             mapper = Mapper.select("c");
128         } else if ("rules".equalsIgnoreCase(partitioning)) {
129             throw new UnsupportedOperationException("Rule-based partitioning not yet implemented");
130         } else if (!"none".equals(partitioning)) {
131             throw new IllegalArgumentException("Unknown partitioning scheme: " + partitioning);
132         }
133 
134         // Read static data, if any
135         final String[] staticSpecs = options.getPositionalArgs(String.class)
136                 .toArray(new String[0]);
137         final RDFSource staticData = staticSpecs.length == 0 ? null : RDFProcessors.track(
138                 new Tracker(LOGGER, null, "%d static triples read (%d tr/s avg)", //
139                         "%d static triples read (%d tr/s, %d tr/s avg)")).wrap(
140                 RDFSources.read(true, preserveBNodes, base, null, staticSpecs));
141 
142         // Build processor
143         return new RuleProcessor(ruleset, mapper, dropBNodeTypes, staticData, emitStatic,
144                 staticContext);
145     }

147     public RuleProcessor(final Ruleset ruleset, @Nullable final Mapper mapper,
148             final boolean dropBNodeTypes) {
149         this(ruleset, mapper, dropBNodeTypes, null, false, null);

152     public RuleProcessor(final Ruleset ruleset, @Nullable final Mapper mapper,
153             final boolean dropBNodeTypes, @Nullable final RDFSource staticData,
154             final boolean emitStatic, @Nullable final URI staticContext) {
155 
156         // Setup the handler receiving closure of static data, if any.
157         final List<Statement> staticClosure = new ArrayList<>();
158         RDFHandler staticSink = RDFHandlers.NIL;
159         if (staticData != null && emitStatic) {
160             staticSink = RDFHandlers.wrap(staticClosure);
161             if (staticContext != null) {
162                 final URI uri = staticContext.equals(SESAME.NIL) ? null : staticContext;
163                 staticSink = new AbstractRDFHandlerWrapper(staticSink) {
164 
165                     @Override
166                     public void handleStatement(final Statement stmt) throws RDFHandlerException {
167                         super.handleStatement(Statements.VALUE_FACTORY.createStatement(
168                                 stmt.getSubject(), stmt.getPredicate(), stmt.getObject(), uri));
169                     }
170 
171                 };
172             }
173         }
174 
175         // Build the dynamic rule engine by closing static data and proprocessing rules
176         final RuleEngine engine = preprocess(ruleset, staticData, staticSink);
177 
178         // Setup object
179         this.engine = engine;
180         this.mapper = mapper;
181         this.staticClosure = RDFSources.wrap(staticClosure);
182         this.dropBNodeTypes = dropBNodeTypes;
183     }

185     private static RuleEngine preprocess(final Ruleset ruleset,
186             @Nullable final RDFSource staticData, final RDFHandler staticSink) {
187 
188         // Build the dynamic engine and compute the static closure by handling two cases
189         LOGGER.info("Processing {} rules {} static data", ruleset.getRules().size(),
190                 staticData == null ? "without" : "with");
191         final long ts = System.currentTimeMillis();
192         if (staticData == null) {
193 
194             // (1) Static data not provided, use input rules without static/dynamic distinction
195             final Ruleset mergedRuleset = ruleset.transformMergeHeads();
196             final RuleEngine.Builder builder = RuleEngine.builder(null);
197             for (final Rule rule : mergedRuleset.getRules()) {
198                 builder.addRule(rule.getID().stringValue(), rule.getHead(), rule.getBody());
199             }
200             final RuleEngine engine = builder.build();
201             LOGGER.info("Rule engine initialized with {} rules in {} ms", mergedRuleset.getRules()
202                     .size(), System.currentTimeMillis() - ts);
203             return engine;
204 
205         } else {
206 
207             // (2) Static data provided, use dynamic rules obtained by preprocessing.
208             // First build the static rule engine
209             final RuleEngine.Builder staticBuilder = RuleEngine.builder(null);
210             for (final Rule rule : ruleset.transformMergeHeads().getRules()) {
211                 staticBuilder.addRule(rule.getID().stringValue() + "__static", rule.getHead(),
212                         rule.getBody());
213             }
214             for (final Rule rule : ruleset.getPreprocessingRuleset().getRules()) {
215                 staticBuilder.addRule(rule.getID().stringValue(), rule.getHead(), rule.getBody());
216             }
217             final RuleEngine staticEngine = staticBuilder.build();
218 
219             // Then run the static engine to compute the closure of static data and build the
220             // dynamic ruleset based on bindings computed for preprocessing rules.
221             final Map<URI, List<BindingSet>> bindings = new HashMap<>();
222             final Map<String, List<BindingSet>> bindingsHelper = new HashMap<>();
223             for (final Rule rule : ruleset.getPreprocessingRuleset().getRules()) {
224                 final List<BindingSet> list = new ArrayList<>();
225                 bindings.put(rule.getID(), list);
226                 bindingsHelper.put(rule.getID().stringValue(), list);
227             }
228             final RDFHandler handler = staticEngine.newSession(staticSink, new Callback() {
229 
230                 @Override
231                 public boolean ruleTriggered(final RDFHandler handler, final String id,
232                         final BindingSet bindings) {
233                     final List<BindingSet> list = bindingsHelper.get(id);
234                     if (list != null) {
235                         list.add(bindings);
236                     }
237                     return true;
238                 }
239 
240             });
241             try {
242                 staticData.emit(handler, 1);
243             } catch (final RDFHandlerException ex) {
244                 throw new RuntimeException(ex);
245             }
246             final Ruleset dynamicRuleset = ruleset.getDynamicRuleset(bindings)
247                     .transformMergeHeads();
248 
249             // Build the dynamic engine
250             final RuleEngine.Builder dynamicBuilder = RuleEngine.builder(null);
251             for (final Rule rule : dynamicRuleset.getRules()) {
252                 dynamicBuilder.addRule(rule.getID().stringValue(), rule.getHead(), rule.getBody());
253             }
254             final RuleEngine engine = dynamicBuilder.build();
255 
256             // Log results
257             LOGGER.info("Rule engine initialized with {} dynamic rules in {} ms", dynamicRuleset
258                     .getRules().size(), System.currentTimeMillis() - ts);
259             return engine;
260         }
261     }

263     @Override
264     public RDFHandler wrap(final RDFHandler handler) {
265 
266         // Start from the supplied handler
267         RDFHandler result = handler;
268 
269         // If necessary, filter the handler so to drop <s rdf:type _:bnode> statements
270         if (this.dropBNodeTypes) {
271             result = new AbstractRDFHandlerWrapper(result) {
272 
273                 @Override
274                 public void handleStatement(final Statement stmt) throws RDFHandlerException {
275                     if (!RDF.TYPE.equals(stmt.getPredicate())
276                             || !(stmt.getObject() instanceof BNode)) {
277                         super.handleStatement(stmt);
278                     }
279                 }
280 
281             };
282         }
283 
284         // If necessary, filter the handler so to inject the static closure (in parallel)
285         if (this.staticClosure != null) {
286             result = RDFProcessors.inject(this.staticClosure).wrap(result);
287         }
288 
289         // Filter the handler to perform inference. Handle two cases.
290         if (this.mapper == null) {
291 
292             // (1) No mapper: just invoke the rule engine
293             result = this.engine.newSession(result);
294 
295         } else {
296 
297             // (2) Mapper configured: perform map/reduce and do inference on reduce phase
298             result = RDFProcessors.mapReduce(this.mapper, new Reducer() {
299 
300                 @Override
301                 public void reduce(final Value key, final Statement[] stmts,
302                         final RDFHandler handler) throws RDFHandlerException {
303                     final RDFHandler session = RuleProcessor.this.engine.newSession(RDFHandlers
304                             .ignoreMethods(handler, RDFHandlers.METHOD_START_RDF
305                                     | RDFHandlers.METHOD_END_RDF | RDFHandlers.METHOD_CLOSE));
306                     try {
307                         session.startRDF();
308                         for (final Statement stmt : stmts) {
309                             session.handleStatement(stmt);
310                         }
311                         session.endRDF();
312                     } finally {
313                         IO.closeQuietly(session);
314                     }
315                 }
316 
317             }, true).wrap(result);
318 
319         }
320 
321         // Return the resulting handler after all the necessary wrappings
322         return result;
323     }
