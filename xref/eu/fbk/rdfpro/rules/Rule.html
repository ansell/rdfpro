29  public final class Rule {
30  
31      private final URI id;
32  
33      @Nullable
34      private final TupleExpr head;
35  
36      @Nullable
37      private final TupleExpr body;
38  
39      public Rule(final URI id, @Nullable final TupleExpr head, @Nullable final TupleExpr body) {
40          this.id = id;
41          this.head = head;
42          this.body = body;
43      }
44  
45      public URI getID() {
46          return this.id;
47      }
48  
49      public TupleExpr getHead() {
50          return this.head;
51      }
52  
53      public TupleExpr getBody() {
54          return this.body;
55      }
56  
57      @Override
58      public boolean equals(final Object object) {
59          if (object == this) {
60              return true;
61          }
62          if (!(object instanceof Rule)) {
63              return false;
64          }
65          final Rule other = (Rule) object;
66          return this.id.equals(other.id);
67      }
68  
69      @Override
70      public int hashCode() {
71          return this.id.hashCode();
72      }
73  
74      @Override
75      public String toString() {
76          try {
77              String head = "<empty>";
78              String body = "<empty>";
79              if (this.head != null) {
80                  head = new SPARQLRenderer(Namespaces.DEFAULT.prefixMap(), false)
81                          .renderTupleExpr(this.head);
82              }
83              if (this.body != null) {
84                  body = new SPARQLRenderer(Namespaces.DEFAULT.prefixMap(), false)
85                          .renderTupleExpr(this.body);
86              }
87              final String id = this.id instanceof BNode ? ((BNode) this.id).getID() : this.id
88                      .getLocalName();
89              return id + ": " + head.replaceAll("[\n\r\t ]+", " ") + " :- "
90                      + body.replaceAll("[\n\r\t ]+", " ");
91          } catch (final Exception ex) {
92              throw new RuntimeException(ex);
93          }
94      }
95  
96      public <T extends Collection<? super Statement>> T toRDF(final T output) {
97  
98          // Emit statements ID a rr:Rule; rr:head HEAD; rr:body BODY.
99          final ValueFactory vf = Statements.VALUE_FACTORY;
100         output.add(vf.createStatement(this.id, RDF.TYPE, RR.RULE));
101         try {
102             if (this.head != null) {
103                 output.add(vf.createStatement(this.id, RR.HEAD,
104                         vf.createLiteral(new SparqlTupleExprRenderer().render(this.head))));
105             }
106             if (this.body != null) {
107                 output.add(vf.createStatement(this.id, RR.BODY,
108                         vf.createLiteral(new SparqlTupleExprRenderer().render(this.body))));
109             }
110         } catch (final Exception ex) {
111             throw new RuntimeException(ex);
112         }
113         return output;
114     }
115 
116     public static List<Rule> fromRDF(final Iterable<Statement> model) {
117 
118         // Load namespaces from model metadata, reusing default prefix/ns mappings
119         final Map<String, String> namespaces = new HashMap<>(Namespaces.DEFAULT.uriMap());
120         if (model instanceof Model) {
121             for (final Namespace namespace : ((Model) model).getNamespaces()) {
122                 namespaces.put(namespace.getPrefix(), namespace.getName());
123             }
124         }
125         for (final Statement stmt : model) {
126             if (stmt.getSubject() instanceof URI && stmt.getObject() instanceof Literal
127                     && stmt.getPredicate().equals(RR.PREFIX_PROPERTY)) {
128                 namespaces.put(stmt.getObject().stringValue(), stmt.getSubject().stringValue());
129             }
130         }
131 
132         // Extract ids, heads and bodies from ruleset data
133         final Set<URI> ids = new TreeSet<>(Statements.valueComparator());
134         final Map<URI, TupleExpr> heads = new HashMap<>();
135         final Map<URI, TupleExpr> bodies = new HashMap<>();
136         for (final Statement stmt : model) {
137             try {
138                 if (stmt.getSubject() instanceof URI) {
139                     final URI pred = stmt.getPredicate();
140                     final Map<URI, TupleExpr> map = pred.equals(RR.HEAD) ? heads : pred
141                             .equals(RR.BODY) ? bodies : null;
142                     if (map != null) {
143                         ids.add((URI) stmt.getSubject());
144                         map.put((URI) stmt.getSubject(), Algebra.parseTupleExpr(stmt.getObject()
145                                 .stringValue(), null, namespaces));
146                     }
147                 }
148             } catch (final Throwable ex) {
149                 throw new IllegalArgumentException(
150                         "Could not parse rule head/body from statement: " + stmt, ex);
151             }
152         }
153 
154         // Generate the rules from parsed heads and bodies
155         final List<Rule> rules = new ArrayList<>();
156         for (final URI id : ids) {
157             final TupleExpr head = heads.get(id);
158             final TupleExpr body = bodies.get(id);
159             rules.add(new Rule(id, head, body));
160         }
161 
162         return rules;
163     }
