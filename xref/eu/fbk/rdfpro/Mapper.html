14  package eu.fbk.rdfpro;
15  
16  import org.openrdf.model.BNode;
17  import org.openrdf.model.Literal;
18  import org.openrdf.model.Statement;
19  import org.openrdf.model.URI;
20  import org.openrdf.model.Value;
21  import org.openrdf.model.impl.URIImpl;
22  import org.openrdf.rio.RDFHandlerException;
23  
24  import eu.fbk.rdfpro.util.Hash;
25  import eu.fbk.rdfpro.util.Statements;
26  
27  /**
28   * Mapper function in a MapReduce job.
29   * <p>
30   * A {@code Mapper} object is used in a MapReduce job (see
31   * {@link RDFProcessors#mapReduce(Mapper, Reducer, boolean)}) for mapping a statement to zero or
32   * more {@code Value keys}; statements are then grouped by key and each partition is processed in
33   * a reduce job (see {@link Reducer}) to produce the final output statements. Mapping is done by
34   * method {@link #map(Statement)}. It can return zero keys to drop the statement, or a key array
35   * including {@link #BYPASS_KEY} to force the statement to be directly emitted in output,
36   * bypassing the reduce stage.
37   * </p>
38   * <p>
39   * A common type of {@code Mapper} is the one extracting a component or hashing a subset of
40   * components of the input statement. This mapper is already implemented and can be instantiated
41   * using the factory method {@link #select(String)}.
42   * </p>
43   * <p>
44   * Implementations of this interface should be thread-safe, as method {@code map()} is meant to be
45   * invoked concurrently by different threads on different statements.
46   * </p>
47   */
48  @FunctionalInterface
49  public interface Mapper {
50  
51      /** Special key used to bypass the reduce stage and directly emit the statement in output. */
52      URI BYPASS_KEY = new URIImpl("rdfpro:bypass");
53  
54      /**
55       * Maps a statement to zero or more {@code Value} keys. When used in a MapReduce job,
56       * returning zero keys has the effect of dropping the statements, otherwise the statement will
57       * be put in each key partition, each one being later processed in a reduce job (see
58       * {@link Reducer}); if one of the keys is {@link #BYPASS_KEY}, then the statement will be
59       * (also) directly emitted in output, bypassing the reduce stage (this can be used to apply
60       * the MapReduce paradigm to only a subset of statements). Null keys are admissible in output
61       * (e.g., to denote the default context).
62       *
63       * @param statement
64       *            the statement to map, not null
65       * @return an array with zero or more {@code Value} keys associated to the statement, not null
66       * @throws RDFHandlerException
67       *             on error
68       */
69      Value[] map(Statement statement) throws RDFHandlerException;
70  
71      /**
72       * Returns a {@code Mapper} returning a single key based on one or more selected components of
73       * the input statement. Parameter {@code components} is a {@code s} , {@code p}, {@code o},
74       * {@code c} string specifying which components should be selected. If a single component is
75       * selected it is returned as the key unchanged. If multiple components are selected, they are
76       * merged and hashed to produce the returned key. In any case, exactly one key is returned for
77       * each input statement.
78       *
79       * @param components
80       *            a string of symbols {@code s} , {@code p}, {@code o}, {@code c} specifying which
81       *            components to select, not null nor empty
82       * @return the created {@code Mapper}
83       */
84      public static Mapper select(final String components) {
85  
86          final String comp = components.trim().toLowerCase();
87  
88          int num = 0;
89          for (int i = 0; i < comp.length(); ++i) {
90              final char c = comp.charAt(i);
91              final int b = c == 's' ? 0x8 : c == 'p' ? 0x4 : c == 'o' ? 0x2 : c == 'c' ? 0x1 : -1;
92              if (b < 0 || (num & b) != 0) {
93                  throw new IllegalArgumentException("Invalid components '" + components + "'");
94              }
95              num = num | b;
96          }
97          final int mask = num;
98  
99          if (mask == 0x08 || mask == 0x04 || mask == 0x02 || mask == 0x01) {
100             return new Mapper() {
101 
102                 @Override
103                 public Value[] map(final Statement statement) throws RDFHandlerException {
104                     switch (mask) {
105                     case 0x08:
106                         return new Value[] { statement.getSubject() };
107                     case 0x04:
108                         return new Value[] { statement.getPredicate() };
109                     case 0x02:
110                         return new Value[] { statement.getObject() };
111                     case 0x01:
112                         return new Value[] { statement.getContext() };
113                     default:
114                         throw new Error();
115                     }
116                 }
117 
118             };
119         }
120 
121         return new Mapper() {
122 
123             private final boolean hasSubj = (mask & 0x80) != 0;
124 
125             private final boolean hasPred = (mask & 0x40) != 0;
126 
127             private final boolean hasObj = (mask & 0x20) != 0;
128 
129             private final boolean hasCtx = (mask & 0x10) != 0;
130 
131             @Override
132             public Value[] map(final Statement statement) throws RDFHandlerException {
133 
134                 int header = 0;
135                 int count = 0;
136 
137                 if (hasSubj) {
138                     final int bits = classify(statement.getSubject());
139                     header |= bits << 24;
140                     count += bits & 0xF;
141                 }
142                 if (hasPred) {
143                     final int bits = classify(statement.getPredicate());
144                     header |= bits << 16;
145                     count += bits & 0xF;
146                 }
147                 if (hasObj) {
148                     final int bits = classify(statement.getObject());
149                     header |= bits << 8;
150                     count += bits & 0xF;
151                 }
152                 if (hasCtx) {
153                     final int bits = classify(statement.getContext());
154                     header |= bits;
155                     count += bits & 0xF;
156                 }
157 
158                 final String[] strings = new String[count];
159                 int index = 0;
160                 strings[index++] = Integer.toString(header);
161                 if (hasSubj) {
162                     index = add(strings, index, statement.getSubject());
163                 }
164                 if (hasPred) {
165                     index = add(strings, index, statement.getPredicate());
166                 }
167                 if (hasObj) {
168                     index = add(strings, index, statement.getObject());
169                 }
170                 if (hasCtx) {
171                     index = add(strings, index, statement.getContext());
172                 }
173 
174                 final String hash = Hash.murmur3(strings).toString();
175                 return new Value[] { Statements.VALUE_FACTORY.createBNode(hash) };
176             }
177 
178             private int classify(final Value value) {
179                 if (value == null) {
180                     return 0;
181                 } else if (value instanceof BNode) {
182                     return 0x11;
183                 } else if (value instanceof URI) {
184                     return 0x21;
185                 }
186                 final Literal l = (Literal) value;
187                 if (l.getDatatype() != null) {
188                     return 0x42;
189                 } else if (l.getLabel() != null) {
190                     return 0x52;
191                 } else {
192                     return 0x31;
193                 }
194             }
195 
196             private int add(final String[] strings, int index, final Value value) {
197                 if (value instanceof URI || value instanceof BNode) {
198                     strings[index++] = value.stringValue();
199                 } else if (value instanceof Literal) {
200                     final Literal l = (Literal) value;
201                     strings[index++] = l.getLabel();
202                     if (l.getDatatype() != null) {
203                         strings[index++] = l.getDatatype().stringValue();
204                     } else if (l.getLanguage() != null) {
205                         strings[index++] = l.getLanguage();
206                     }
207                 }
208                 return index;
209             }
210 
211         };
212     }
