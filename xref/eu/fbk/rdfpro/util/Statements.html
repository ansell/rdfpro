102         final int index = fileSpec.indexOf(':');
103         if (index > 0 && RDFFormat.forFileName("test." + fileSpec.substring(0, index)) != null) {

109     public static RDFFormat toRDFFormat(final String fileSpec) {
110         final int index = fileSpec.indexOf(':');
111         if (index > 0) {
112             final RDFFormat format = RDFFormat.forFileName("test." + fileSpec.substring(0, index));
113             if (format != null) {
114                 return format;
115             }
116         }
117         final RDFFormat format = RDFFormat.forFileName(fileSpec);
118         if (format == null) {
119             throw new IllegalArgumentException("Unknown RDF format for " + fileSpec);
120         }
121         return format;
122     }

124     public static boolean isRDFFormatTextBased(final RDFFormat format) {
125         for (final String ext : format.getFileExtensions()) {
126             if (ext.equalsIgnoreCase("rdf") || ext.equalsIgnoreCase("rj")
127                     || ext.equalsIgnoreCase("jsonld") || ext.equalsIgnoreCase("nt")
128                     || ext.equalsIgnoreCase("nq") || ext.equalsIgnoreCase("trix")
129                     || ext.equalsIgnoreCase("trig") || ext.equalsIgnoreCase("tql")
130                     || ext.equalsIgnoreCase("ttl") || ext.equalsIgnoreCase("n3")) {
131                 return true;
132             }
133         }
134         return false;
135     }

147     public static Value shortenValue(final Value value, final int threshold) {
148         if (value instanceof Literal) {
149             final Literal literal = (Literal) value;
150             final URI datatype = literal.getDatatype();
151             final String language = literal.getLanguage();
152             final String label = ((Literal) value).getLabel();
153             if (label.length() > threshold
154                     && (datatype == null || datatype.equals(XMLSchema.STRING))) {
155                 int offset = threshold;
156                 for (int i = threshold; i >= 0; --i) {
157                     if (Character.isWhitespace(label.charAt(i))) {
158                         offset = i;
159                         break;
160                     }
161                 }
162                 final String newLabel = label.substring(0, offset) + "...";
163                 if (datatype != null) {
164                     return VALUE_FACTORY.createLiteral(newLabel, datatype);
165                 } else if (language != null) {
166                     return VALUE_FACTORY.createLiteral(newLabel, language);
167                 } else {
168                     return VALUE_FACTORY.createLiteral(newLabel);
169                 }
170             }
171         }
172         return value;
173     }

180     public static String formatValue(@Nullable final Value value,
181             @Nullable final Namespaces namespaces) {
182         if (value == null) {
183             return null;
184         }
185         try {
186             final StringBuilder builder = new StringBuilder(value.stringValue().length() * 2);
187             formatValue(value, namespaces, builder);
188             return builder.toString();
189         } catch (final Throwable ex) {
190             throw new Error("Unexpected exception (!)", ex);
191         }

194     public static void formatValue(final Value value, @Nullable final Namespaces namespaces,
195             final Appendable out) throws IOException {
196         if (value instanceof URI) {
197             formatURI((URI) value, out, namespaces);
198         } else if (value instanceof BNode) {
199             formatBNode((BNode) value, out);
200         } else if (value instanceof Literal) {
201             formatLiteral((Literal) value, out, namespaces);
202         } else {
203             throw new Error("Unexpected value class (!): " + value.getClass().getName());
204         }
205     }

207     private static boolean isGoodQName(final String prefix, final String name) { // good to emit
208 
209         final int prefixLen = prefix.length();
210         if (prefixLen > 0) {
211             if (!isPN_CHARS_BASE(prefix.charAt(0)) || !isPN_CHARS(prefix.charAt(prefixLen - 1))) {
212                 return false;
213             }
214             for (int i = 1; i < prefixLen - 1; ++i) {
215                 final char c = prefix.charAt(i);
216                 if (!isPN_CHARS(c) && c != '.') {
217                     return false;
218                 }
219             }
220         }
221 
222         final int nameLen = name.length();
223         if (nameLen > 0) {
224             int i = 0;
225             while (i < nameLen) {
226                 final char c = name.charAt(i++);
227                 if (!isPN_CHARS_BASE(c) && c != ':' && (i != 1 || c != '_' && !isNumber(c))
228                         && (i == 1 || !isPN_CHARS(c) && (i == nameLen || c != '.'))) {
229                     return false;
230                 }
231             }
232         }
233 
234         return true;
235     }

237     private static void formatURI(final URI uri, final Appendable out,
238             @Nullable final Namespaces namespaces) throws IOException {
239 
240         if (namespaces != null) {
241             final String prefix = namespaces.prefixFor(uri.getNamespace());
242             if (prefix != null) {
243                 final String name = uri.getLocalName();
244                 if (isGoodQName(prefix, name)) {
245                     out.append(prefix);
246                     out.append(":");
247                     out.append(name);
248                     return;
249                 }
250             }
251         }
252 
253         final String string = uri.stringValue();
254         final int len = string.length();
255         out.append('<');
256         for (int i = 0; i < len; ++i) {
257             final char ch = string.charAt(i);
258             switch (ch) {
259             case 0x22: // "
260                 out.append("\\u0022");
261                 break;
262             case 0x3C: // <
263                 out.append("\\u003C");
264                 break;
265             case 0x3E: // >
266                 out.append("\\u003E");
267                 break;
268             case 0x5C: // \
269                 out.append("\\u005C");
270                 break;
271             case 0x5E: // ^
272                 out.append("\\u005E");
273                 break;
274             case 0x60: // `
275                 out.append("\\u0060");
276                 break;
277             case 0x7B: // {
278                 out.append("\\u007B");
279                 break;
280             case 0x7C: // |
281                 out.append("\\u007C");
282                 break;
283             case 0x7D: // }
284                 out.append("\\u007D");
285                 break;
286             case 0x7F: // delete control char (not strictly necessary)
287                 out.append("\\u007F");
288                 break;
289             default:
290                 if (ch <= 32) { // control char and ' '
291                     out.append("\\u00").append(Character.forDigit(ch / 16, 16))
292                             .append(Character.forDigit(ch % 16, 16));
293                 } else {
294                     out.append(ch);
295                 }
296             }
297         }
298         out.append('>');
299     }

301     private static void formatBNode(final BNode bnode, final Appendable out) throws IOException {
302         final String id = bnode.getID();
303         final int last = id.length() - 1;
304         out.append('_').append(':');
305         if (last < 0) {
306             out.append("genid-hash-").append(Integer.toHexString(System.identityHashCode(bnode)));
307         } else {
308             char ch = id.charAt(0);
309             if (isPN_CHARS_U(ch) || isNumber(ch)) {
310                 out.append(ch);
311             } else {
312                 out.append("genid-start-").append(ch);
313             }
314             if (last > 0) {
315                 for (int i = 1; i < last; ++i) {
316                     ch = id.charAt(i);
317                     if (isPN_CHARS(ch) || ch == '.') {
318                         out.append(ch);
319                     } else {
320                         out.append(Integer.toHexString(ch));
321                     }
322                 }
323                 ch = id.charAt(last);
324                 if (isPN_CHARS(ch)) {
325                     out.append(ch);
326                 } else {
327                     out.append(Integer.toHexString(ch));
328                 }
329             }
330         }
331     }

333     private static void formatLiteral(final Literal literal, final Appendable out,
334             @Nullable final Namespaces namespaces) throws IOException {
335         final String label = literal.getLabel();
336         final int length = label.length();
337         out.append('"');
338         for (int i = 0; i < length; ++i) {
339             final char ch = label.charAt(i);
340             switch (ch) {
341             case 0x08: // \b
342                 out.append('\\');
343                 out.append('b');
344                 break;
345             case 0x09: // \t
346                 out.append('\\');
347                 out.append('t');
348                 break;
349             case 0x0A: // \n
350                 out.append('\\');
351                 out.append('n');
352                 break;
353             case 0x0C: // \f
354                 out.append('\\');
355                 out.append('f');
356                 break;
357             case 0x0D: // \r
358                 out.append('\\');
359                 out.append('r');
360                 break;
361             case 0x22: // "
362                 out.append('\\');
363                 out.append('"');
364                 break;
365             case 0x5C: // \
366                 out.append('\\');
367                 out.append('\\');
368                 break;
369             case 0x7F: // delete control char
370                 out.append("\\u007F");
371                 break;
372             default:
373                 if (ch < 32) { // other control char (not strictly necessary)
374                     out.append("\\u00");
375                     out.append(Character.forDigit(ch / 16, 16));
376                     out.append(Character.forDigit(ch % 16, 16));
377                 } else {
378                     out.append(ch);
379                 }
380             }
381         }
382         out.append('"');
383         final URI datatype = literal.getDatatype();
384         if (datatype != null) {
385             out.append('^');
386             out.append('^');
387             formatURI(datatype, out, namespaces);
388         } else {
389             final String language = literal.getLanguage();
390             if (language != null) {
391                 out.append('@');
392                 final int len = language.length();
393                 boolean minusFound = false;
394                 boolean valid = true;
395                 for (int i = 0; i < len; ++i) {
396                     final char ch = language.charAt(i);
397                     if (ch == '-') {
398                         minusFound = true;
399                         if (i == 0) {
400                             valid = false;
401                         } else {
402                             final char prev = language.charAt(i - 1);
403                             valid &= isLetter(prev) || isNumber(prev);
404                         }
405                     } else if (isNumber(ch)) {
406                         valid &= minusFound;
407                     } else {
408                         valid &= isLetter(ch);
409                     }
410                     out.append(ch);
411                 }
412                 if (!valid || language.charAt(len - 1) == '-') {
413                     throw new IllegalArgumentException("Invalid language tag '" + language
414                             + "' in '" + literal + "'");
415                 }
416             }
417         }
418     }

425     public static Value parseValue(@Nullable final CharSequence sequence,
426             @Nullable final Namespaces namespaces) {
427         if (sequence == null) {
428             return null;
429         }
430         final int c = sequence.charAt(0);
431         if (c == '_') {
432             return parseBNode(sequence);
433         } else if (c == '"' || c == '\'') {
434             return parseLiteral(sequence, namespaces);
435         } else {
436             return parseURI(sequence, namespaces);
437         }
438     }

440     private static URI parseURI(final CharSequence sequence, @Nullable final Namespaces namespaces) {
441 
442         if (sequence.charAt(0) == '<') {
443             final int last = sequence.length() - 1;
444             final StringBuilder builder = new StringBuilder(last - 1);
445             if (sequence.charAt(last) != '>') {
446                 throw new IllegalArgumentException("Invalid URI: " + sequence);
447             }
448             int i = 1;
449             while (i < last) {
450                 char c = sequence.charAt(i++);
451                 if (c < 32) { // discard control chars but accept other chars forbidden by W3C
452                     throw new IllegalArgumentException("Invalid char '" + c + "' in URI: "
453                             + sequence);
454                 } else if (c != '\\') {
455                     builder.append(c);
456                 } else {
457                     if (i == last) {
458                         throw new IllegalArgumentException("Invalid URI: " + sequence);
459                     }
460                     c = sequence.charAt(i++);
461                     if (c == 'u') {
462                         builder.append(parseHex(sequence, i, 4));
463                         i += 4;
464                     } else if (c == 'U') {
465                         builder.append(parseHex(sequence, i, 8));
466                         i += 8;
467                     } else {
468                         builder.append(c); // accept \> and \\ plus others
469                     }
470                 }
471             }
472             return VALUE_FACTORY.createURI(builder.toString());
473 
474         } else if (namespaces != null) {
475             final int len = sequence.length();
476             final StringBuilder builder = new StringBuilder(len);
477 
478             int i = 0;
479             while (i < len) {
480                 final char c = sequence.charAt(i++);
481                 if (c == ':') {
482                     if (i > 2 && !isPN_CHARS(sequence.charAt(i - 2))) {
483                         throw new IllegalArgumentException("Invalid qname " + sequence);
484                     }
485                     break;
486                 } else if (i == 1 && !isPN_CHARS_BASE(c) || i > 1 && !isPN_CHARS(c) && c != '.') {
487                     throw new IllegalArgumentException("Invalid qname " + sequence);
488                 } else {
489                     builder.append(c);
490                 }
491             }
492             final String prefix = builder.toString();
493             final String namespace = namespaces.uriFor(prefix);
494             if (namespace == null) {
495                 throw new IllegalArgumentException("Unknown prefix " + prefix);
496             }
497             builder.setLength(0);
498 
499             while (i < len) {
500                 final char c = sequence.charAt(i++);
501                 if (c == '%') {
502                     builder.append(parseHex(sequence, i, 2));
503                     i += 2;
504                 } else if (c == '\\') {
505                     final char d = sequence.charAt(i++);
506                     if (d == '_' || d == '~' || d == '.' || d == '-' || d == '!' || d == '$'
507                             || d == '&' || d == '\'' || d == '(' || d == ')' || d == '*'
508                             || d == '+' || d == ',' || d == ';' || d == '=' || d == '/'
509                             || d == '?' || d == '#' || d == '@' || d == '%') {
510                         builder.append(d);
511                     } else {
512                         throw new IllegalArgumentException("Invalid qname " + sequence);
513                     }
514                 } else if (isPN_CHARS_BASE(c) || c == ':' || i == 1 && (c == '_' || isNumber(c))
515                         || i > 1 && (isPN_CHARS(c) || i < len && c == '.')) {
516                     builder.append(c);
517                 } else {
518                     throw new IllegalArgumentException("Invalid qname " + sequence);
519                 }
520             }
521             final String name = builder.toString();
522 
523             return VALUE_FACTORY.createURI(namespace, name);
524         }
525 
526         throw new IllegalArgumentException("Unsupported qname " + sequence);
527     }

529     private static BNode parseBNode(final CharSequence sequence) {
530         final int len = sequence.length();
531         if (len > 2 && sequence.charAt(1) == ':') {
532             final StringBuilder builder = new StringBuilder(len - 2);
533             boolean ok = true;
534             char c = sequence.charAt(2);
535             builder.append(c);
536             ok &= isPN_CHARS_U(c) || isNumber(c);
537             for (int i = 2; i < len - 1; ++i) {
538                 c = sequence.charAt(i);
539                 builder.append(c);
540                 ok &= isPN_CHARS(c) || c == '.';
541             }
542             c = sequence.charAt(len - 1);
543             builder.append(c);
544             ok &= isPN_CHARS(c);
545             if (ok) {
546                 return VALUE_FACTORY.createBNode(builder.toString());
547             }
548         }
549         throw new IllegalArgumentException("Invalid BNode '" + sequence + "'");
550     }

552     private static Literal parseLiteral(final CharSequence sequence,
553             @Nullable final Namespaces namespaces) {
554 
555         final StringBuilder builder = new StringBuilder(sequence.length());
556         final int len = sequence.length();
557 
558         final char delim = sequence.charAt(0);
559         char c = 0;
560         int i = 1;
561         while (i < len && (c = sequence.charAt(i++)) != delim) {
562             if (c == '\\' && i < len) {
563                 c = sequence.charAt(i++);
564                 switch (c) {
565                 case 'b':
566                     builder.append('\b');
567                     break;
568                 case 'f':
569                     builder.append('\f');
570                     break;
571                 case 'n':
572                     builder.append('\n');
573                     break;
574                 case 'r':
575                     builder.append('\r');
576                     break;
577                 case 't':
578                     builder.append('\t');
579                     break;
580                 case 'u':
581                     builder.append(parseHex(sequence, i, 4));
582                     i += 4;
583                     break;
584                 case 'U':
585                     builder.append(parseHex(sequence, i, 8));
586                     i += 8;
587                     break;
588                 default:
589                     builder.append(c); // handles ' " \
590                     break;
591                 }
592             } else {
593                 builder.append(c);
594             }
595         }
596         final String label = builder.toString();
597 
598         if (i == len && c == delim) {
599             return VALUE_FACTORY.createLiteral(label);
600 
601         } else if (i < len - 2 && sequence.charAt(i) == '^' && sequence.charAt(i + 1) == '^') {
602             final URI datatype = parseURI(sequence.subSequence(i + 2, len), namespaces);
603             return VALUE_FACTORY.createLiteral(label, datatype);
604 
605         } else if (i < len - 1 && sequence.charAt(i) == '@') {
606             builder.setLength(0);
607             boolean minusFound = false;
608             for (int j = i + 1; j < len; ++j) {
609                 c = sequence.charAt(j);
610                 if (!isLetter(c) && (c != '-' || j == i + 1 || j == len - 1)
611                         && (!isNumber(c) || !minusFound)) {
612                     throw new IllegalArgumentException("Invalid lang in '" + sequence + "'");
613                 }
614                 minusFound |= c == '-';
615                 builder.append(c);
616             }
617             return VALUE_FACTORY.createLiteral(label, builder.toString());
618         }
619 
620         throw new IllegalArgumentException("Invalid literal '" + sequence + "'");
621     }

623     private static char parseHex(final CharSequence sequence, final int index, final int count) {
624         int code = 0;
625         final int len = sequence.length();
626         if (index + count >= len) {
627             throw new IllegalArgumentException("Incomplete hex code '"
628                     + sequence.subSequence(index, len) + "' in RDF value '" + sequence + "'");
629         }
630         for (int i = 0; i < count; ++i) {
631             final char c = sequence.charAt(index + i);
632             final int digit = Character.digit(c, 16);
633             if (digit < 0) {
634                 throw new IllegalArgumentException("Invalid hex digit '" + c + "' in RDF value '"
635                         + sequence + "'");
636             }
637             code = code * 16 + digit;
638         }
639         return (char) code;
640     }

642     private static boolean isPN_CHARS(final int c) { // ok
643         return isPN_CHARS_U(c) || isNumber(c) || c == '-' || c == 0x00B7 || c >= 0x0300
644                 && c <= 0x036F || c >= 0x203F && c <= 0x2040;

713      * <td>{@link Statement}, {@link String}</td>
714      * </tr>
715      * <tr>
716      * <td>{@link Record}</td>
717      * <td>{@link Record}, {@link URI} (ID extracted), {@link String}</td>
718      * </tr>

732      *             specified

734     @SuppressWarnings("unchecked")
735     @Nullable
736     public static <T> T convert(@Nullable final Object object, final Class<T> clazz)
737             throws IllegalArgumentException {
738         if (object == null) {
739             Objects.requireNonNull(clazz);
740             return null;
741         }
742         if (clazz.isInstance(object)) {
743             return (T) object;
744         }
745         final T result = (T) convertObject(object, clazz);
746         if (result != null) {
747             return result;
748         }
749         throw new IllegalArgumentException("Unsupported conversion of " + object + " to " + clazz);

752     /**
753      * General conversion facility, with fall back to default value. This method operates as
754      * {@link #convert(Object, Class)}, but in case the input is null or conversion is not
755      * supported returns the specified default value.
756      *
757      * @param object
758      *            the object to convert, possibly null
759      * @param clazz
760      *            the class to convert to, not null
761      * @param defaultValue
762      *            the default value to fall back to
763      * @param <T>
764      *            the type of result
765      * @return the result of the conversion, or the default value if {@code object} was null,
766      *         conversion failed or is unsupported
767      */
768     @SuppressWarnings("unchecked")
769     @Nullable
770     public static <T> T convert(@Nullable final Object object, final Class<T> clazz,
771             @Nullable final T defaultValue) {
772         if (object == null) {
773             Objects.requireNonNull(clazz);
774             return defaultValue;
775         }
776         if (clazz.isInstance(object)) {
777             return (T) object;
778         }
779         try {
780             final T result = (T) convertObject(object, clazz);
781             return result != null ? result : defaultValue;
782         } catch (final RuntimeException ex) {
783             return defaultValue;
784         }
785     }

787     @Nullable
788     private static Object convertObject(final Object object, final Class<?> clazz) {
789         if (object instanceof Literal) {
790             return convertLiteral((Literal) object, clazz);
791         } else if (object instanceof URI) {
792             return convertURI((URI) object, clazz);
793         } else if (object instanceof String) {
794             return convertString((String) object, clazz);
795         } else if (object instanceof Number) {
796             return convertNumber((Number) object, clazz);
797         } else if (object instanceof Boolean) {
798             return convertBoolean((Boolean) object, clazz);
799         } else if (object instanceof XMLGregorianCalendar) {
800             return convertCalendar((XMLGregorianCalendar) object, clazz);
801         } else if (object instanceof BNode) {
802             return convertBNode((BNode) object, clazz);
803         } else if (object instanceof Statement) {
804             return convertStatement((Statement) object, clazz);
805         } else if (object instanceof GregorianCalendar) {
806             final XMLGregorianCalendar calendar = DATATYPE_FACTORY
807                     .newXMLGregorianCalendar((GregorianCalendar) object);
808             return clazz == XMLGregorianCalendar.class ? calendar : convertCalendar(calendar,
809                     clazz);
810         } else if (object instanceof Date) {
811             final GregorianCalendar calendar = new GregorianCalendar();
812             calendar.setTime((Date) object);
813             final XMLGregorianCalendar xmlCalendar = DATATYPE_FACTORY
814                     .newXMLGregorianCalendar(calendar);
815             return clazz == XMLGregorianCalendar.class ? xmlCalendar : convertCalendar(
816                     xmlCalendar, clazz);
817         } else if (object instanceof Enum<?>) {
818             return convertEnum((Enum<?>) object, clazz);
819         } else if (object instanceof File) {
820             return convertFile((File) object, clazz);
821         }
822         return null;
823     }

833     @Nullable
834     private static Object convertLiteral(final Literal literal, final Class<?> clazz) {
835         final URI datatype = literal.getDatatype();
836         if (datatype == null || datatype.equals(XMLSchema.STRING)) {
837             return convertString(literal.getLabel(), clazz);
838         } else if (datatype.equals(XMLSchema.BOOLEAN)) {
839             return convertBoolean(literal.booleanValue(), clazz);
840         } else if (datatype.equals(XMLSchema.DATE) || datatype.equals(XMLSchema.DATETIME)) {
841             return convertCalendar(literal.calendarValue(), clazz);
842         } else if (datatype.equals(XMLSchema.INT)) {
843             return convertNumber(literal.intValue(), clazz);
844         } else if (datatype.equals(XMLSchema.LONG)) {
845             return convertNumber(literal.longValue(), clazz);
846         } else if (datatype.equals(XMLSchema.DOUBLE)) {
847             return convertNumber(literal.doubleValue(), clazz);
848         } else if (datatype.equals(XMLSchema.FLOAT)) {
849             return convertNumber(literal.floatValue(), clazz);
850         } else if (datatype.equals(XMLSchema.SHORT)) {
851             return convertNumber(literal.shortValue(), clazz);
852         } else if (datatype.equals(XMLSchema.BYTE)) {
853             return convertNumber(literal.byteValue(), clazz);
854         } else if (datatype.equals(XMLSchema.DECIMAL)) {
855             return convertNumber(literal.decimalValue(), clazz);
856         } else if (datatype.equals(XMLSchema.INTEGER)) {
857             return convertNumber(literal.integerValue(), clazz);
858         } else if (datatype.equals(XMLSchema.NON_NEGATIVE_INTEGER)
859                 || datatype.equals(XMLSchema.NON_POSITIVE_INTEGER)
860                 || datatype.equals(XMLSchema.NEGATIVE_INTEGER)
861                 || datatype.equals(XMLSchema.POSITIVE_INTEGER)) {
862             return convertNumber(literal.integerValue(), clazz); // infrequent integer cases
863         } else if (datatype.equals(XMLSchema.NORMALIZEDSTRING) || datatype.equals(XMLSchema.TOKEN)
864                 || datatype.equals(XMLSchema.NMTOKEN) || datatype.equals(XMLSchema.LANGUAGE)
865                 || datatype.equals(XMLSchema.NAME) || datatype.equals(XMLSchema.NCNAME)) {
866             return convertString(literal.getLabel(), clazz); // infrequent string cases
867         }
868         return null;
869     }

883     @Nullable
884     private static Object convertString(final String string, final Class<?> clazz) {
885         if (clazz.isInstance(string)) {
886             return string;
887         } else if (clazz.isAssignableFrom(Literal.class)) {
888             return VALUE_FACTORY.createLiteral(string, XMLSchema.STRING);
889         } else if (clazz.isAssignableFrom(URI.class)) {
890             return VALUE_FACTORY.createURI(string);
891         } else if (clazz.isAssignableFrom(BNode.class)) {
892             return VALUE_FACTORY.createBNode(string.startsWith("_:") ? string.substring(2)
893                     : string);
894         } else if (clazz == Boolean.class || clazz == boolean.class) {
895             return Boolean.valueOf(string);
896         } else if (clazz == Integer.class || clazz == int.class) {
897             return Integer.valueOf((int) toLong(string));
898         } else if (clazz == Long.class || clazz == long.class) {
899             return Long.valueOf(toLong(string));
900         } else if (clazz == Double.class || clazz == double.class) {
901             return Double.valueOf(string);
902         } else if (clazz == Float.class || clazz == float.class) {
903             return Float.valueOf(string);
904         } else if (clazz == Short.class || clazz == short.class) {
905             return Short.valueOf((short) toLong(string));
906         } else if (clazz == Byte.class || clazz == byte.class) {
907             return Byte.valueOf((byte) toLong(string));
908         } else if (clazz == BigDecimal.class) {
909             return new BigDecimal(string);
910         } else if (clazz == BigInteger.class) {
911             return new BigInteger(string);
912         } else if (clazz == AtomicInteger.class) {
913             return new AtomicInteger(Integer.parseInt(string));
914         } else if (clazz == AtomicLong.class) {
915             return new AtomicLong(Long.parseLong(string));
916         } else if (clazz == Date.class) {
917             final String fixed = string.contains("T") ? string : string + "T00:00:00";
918             return DATATYPE_FACTORY.newXMLGregorianCalendar(fixed).toGregorianCalendar().getTime();
919         } else if (clazz.isAssignableFrom(GregorianCalendar.class)) {
920             final String fixed = string.contains("T") ? string : string + "T00:00:00";
921             return DATATYPE_FACTORY.newXMLGregorianCalendar(fixed).toGregorianCalendar();
922         } else if (clazz.isAssignableFrom(XMLGregorianCalendar.class)) {
923             final String fixed = string.contains("T") ? string : string + "T00:00:00";
924             return DATATYPE_FACTORY.newXMLGregorianCalendar(fixed);
925         } else if (clazz == Character.class || clazz == char.class) {
926             return string.isEmpty() ? null : string.charAt(0);
927         } else if (clazz.isEnum()) {
928             for (final Object constant : clazz.getEnumConstants()) {
929                 if (string.equalsIgnoreCase(((Enum<?>) constant).name())) {
930                     return constant;
931                 }
932             }
933             throw new IllegalArgumentException("Illegal " + clazz.getSimpleName() + " constant: "
934                     + string);
935         } else if (clazz == File.class) {
936             return new File(string);
937         }
938         return null;
939     }

973         } else if (clazz == Byte.class || clazz == byte.class) {

976             return toBigDecimal(number);

981         } else if (clazz == AtomicLong.class) {

1077     private static long toLong(final String string) {
1078         long multiplier = 1;
1079         final char c = string.charAt(string.length() - 1);
1080         if (c == 'k' || c == 'K') {
1081             multiplier = 1024;
1082         } else if (c == 'm' || c == 'M') {
1083             multiplier = 1024 * 1024;
1084         } else if (c == 'g' || c == 'G') {
1085             multiplier = 1024 * 1024 * 1024;
1086         }
1087         return Long.parseLong(multiplier == 1 ? string : string.substring(0, string.length() - 1))
1088                 * multiplier;
1089     }
