143         final int index = fileSpec.indexOf(':');
144         if (index > 0) {
145             final String name = "test." + fileSpec.substring(0, index);
146             if (Rio.getParserFormatForFileName(name) != null

154     public static RDFFormat toRDFFormat(final String fileSpec) {
155         final int index = fileSpec.indexOf(':');
156         RDFFormat format = null;
157         if (index > 0) {
158             final String name = "test." + fileSpec.substring(0, index);
159             format = Rio.getParserFormatForFileName(name);
160             if (format == null) {
161                 format = Rio.getWriterFormatForFileName(name);
162             }
163         }
164         if (format == null) {
165             format = Rio.getParserFormatForFileName(fileSpec);
166             if (format == null) {
167                 format = Rio.getWriterFormatForFileName(fileSpec);
168             }
169         }
170         if (format == null) {
171             throw new IllegalArgumentException("Unknown RDF format for " + fileSpec);
172         }
173         return format;
174     }

199     public static Value shortenValue(final Value value, final int threshold) {
200         if (value instanceof Literal) {
201             final Literal literal = (Literal) value;
202             final URI datatype = literal.getDatatype();
203             final String language = literal.getLanguage();
204             final String label = ((Literal) value).getLabel();
205             if (label.length() > threshold
206                     && (datatype == null || datatype.equals(XMLSchema.STRING))) {
207                 int offset = threshold;
208                 for (int i = threshold; i >= 0; --i) {
209                     if (Character.isWhitespace(label.charAt(i))) {
210                         offset = i;
211                         break;
212                     }
213                 }
214                 final String newLabel = label.substring(0, offset) + "...";
215                 if (language != null) {
216                     return VALUE_FACTORY.createLiteral(newLabel, language);
217                 } else if (datatype != null) {
218                     return VALUE_FACTORY.createLiteral(newLabel, datatype);
219                 } else {
220                     return VALUE_FACTORY.createLiteral(newLabel);
221                 }
222             }
223         }
224         return value;
225     }

246     public static void formatValue(final Value value, @Nullable final Namespaces namespaces,
247             final Appendable out) throws IOException {
248         if (value instanceof URI) {
249             formatURI((URI) value, out, namespaces);
250         } else if (value instanceof BNode) {
251             formatBNode((BNode) value, out);

254         } else {
255             throw new Error("Unexpected value class (!): " + value.getClass().getName());
256         }

259     private static boolean isGoodQName(final String prefix, final String name) { // good to emit
260 
261         final int prefixLen = prefix.length();
262         if (prefixLen > 0) {
263             if (!isPN_CHARS_BASE(prefix.charAt(0)) || !isPN_CHARS(prefix.charAt(prefixLen - 1))) {
264                 return false;
265             }
266             for (int i = 1; i < prefixLen - 1; ++i) {
267                 final char c = prefix.charAt(i);
268                 if (!isPN_CHARS(c) && c != '.') {
269                     return false;
270                 }
271             }
272         }
273 
274         final int nameLen = name.length();
275         if (nameLen > 0) {
276             int i = 0;
277             while (i < nameLen) {
278                 final char c = name.charAt(i++);
279                 if (!isPN_CHARS_BASE(c) && c != ':' && (i != 1 || c != '_' && !isNumber(c))
280                         && (i == 1 || !isPN_CHARS(c) && (i == nameLen || c != '.'))) {
281                     return false;
282                 }
283             }
284         }
285 
286         return true;
287     }

289     private static void formatURI(final URI uri, final Appendable out,
290             @Nullable final Namespaces namespaces) throws IOException {
291 
292         if (namespaces != null) {
293             final String prefix = namespaces.prefixFor(uri.getNamespace());
294             if (prefix != null) {
295                 final String name = uri.getLocalName();
296                 if (isGoodQName(prefix, name)) {
297                     out.append(prefix);
298                     out.append(":");
299                     out.append(name);
300                     return;
301                 }
302             }
303         }
304 
305         final String string = uri.stringValue();
306         final int len = string.length();
307         out.append('<');
308         for (int i = 0; i < len; ++i) {
309             final char ch = string.charAt(i);
310             switch (ch) {
311             case 0x22: // "
312                 out.append("\\u0022");
313                 break;
314             case 0x3C: // <
315                 out.append("\\u003C");
316                 break;
317             case 0x3E: // >
318                 out.append("\\u003E");
319                 break;
320             case 0x5C: // \
321                 out.append("\\u005C");
322                 break;
323             case 0x5E: // ^
324                 out.append("\\u005E");
325                 break;
326             case 0x60: // `
327                 out.append("\\u0060");
328                 break;
329             case 0x7B: // {
330                 out.append("\\u007B");
331                 break;
332             case 0x7C: // |
333                 out.append("\\u007C");
334                 break;
335             case 0x7D: // }
336                 out.append("\\u007D");
337                 break;
338             case 0x7F: // delete control char (not strictly necessary)
339                 out.append("\\u007F");
340                 break;
341             default:
342                 if (ch <= 32) { // control char and ' '
343                     out.append("\\u00").append(Character.forDigit(ch / 16, 16))
344                             .append(Character.forDigit(ch % 16, 16));
345                 } else {
346                     out.append(ch);
347                 }
348             }
349         }
350         out.append('>');
351     }

353     private static void formatBNode(final BNode bnode, final Appendable out) throws IOException {
354         final String id = bnode.getID();
355         final int last = id.length() - 1;
356         out.append('_').append(':');
357         if (last < 0) {
358             out.append("genid-hash-").append(Integer.toHexString(System.identityHashCode(bnode)));
359         } else {
360             char ch = id.charAt(0);
361             if (isPN_CHARS_U(ch) || isNumber(ch)) {
362                 out.append(ch);
363             } else {
364                 out.append("genid-start-").append(ch);
365             }
366             if (last > 0) {
367                 for (int i = 1; i < last; ++i) {
368                     ch = id.charAt(i);
369                     if (isPN_CHARS(ch) || ch == '.') {
370                         out.append(ch);
371                     } else {
372                         out.append(Integer.toHexString(ch));
373                     }
374                 }
375                 ch = id.charAt(last);
376                 if (isPN_CHARS(ch)) {
377                     out.append(ch);
378                 } else {
379                     out.append(Integer.toHexString(ch));
380                 }
381             }
382         }
383     }

385     private static void formatLiteral(final Literal literal, final Appendable out,
386             @Nullable final Namespaces namespaces) throws IOException {
387         final String label = literal.getLabel();
388         final int length = label.length();
389         out.append('"');
390         for (int i = 0; i < length; ++i) {
391             final char ch = label.charAt(i);
392             switch (ch) {
393             case 0x08: // \b
394                 out.append('\\');
395                 out.append('b');
396                 break;
397             case 0x09: // \t
398                 out.append('\\');
399                 out.append('t');
400                 break;
401             case 0x0A: // \n
402                 out.append('\\');
403                 out.append('n');
404                 break;
405             case 0x0C: // \f
406                 out.append('\\');
407                 out.append('f');
408                 break;
409             case 0x0D: // \r
410                 out.append('\\');
411                 out.append('r');
412                 break;
413             case 0x22: // "
414                 out.append('\\');
415                 out.append('"');
416                 break;
417             case 0x5C: // \
418                 out.append('\\');
419                 out.append('\\');
420                 break;
421             case 0x7F: // delete control char
422                 out.append("\\u007F");
423                 break;
424             default:
425                 if (ch < 32) { // other control char (not strictly necessary)
426                     out.append("\\u00");
427                     out.append(Character.forDigit(ch / 16, 16));
428                     out.append(Character.forDigit(ch % 16, 16));
429                 } else {
430                     out.append(ch);
431                 }
432             }
433         }
434         out.append('"');
435         final String language = literal.getLanguage();
436         if (language != null) {
437             out.append('@');
438             final int len = language.length();
439             boolean minusFound = false;
440             boolean valid = true;
441             for (int i = 0; i < len; ++i) {
442                 final char ch = language.charAt(i);
443                 if (ch == '-') {
444                     minusFound = true;
445                     if (i == 0) {
446                         valid = false;
447                     } else {
448                         final char prev = language.charAt(i - 1);
449                         valid &= isLetter(prev) || isNumber(prev);
450                     }
451                 } else if (isNumber(ch)) {
452                     valid &= minusFound;
453                 } else {
454                     valid &= isLetter(ch);
455                 }
456                 out.append(ch);
457             }
458             if (!valid || language.charAt(len - 1) == '-') {
459                 throw new IllegalArgumentException("Invalid language tag '" + language + "' in '"
460                         + literal + "'");
461             }
462         } else {
463             final URI datatype = literal.getDatatype();
464             if (datatype != null && !XMLSchema.STRING.equals(datatype)) {
465                 out.append('^');
466                 out.append('^');
467                 formatURI(datatype, out, namespaces);
468             }
469         }
470     }

477     public static Value parseValue(@Nullable final CharSequence sequence,
478             @Nullable final Namespaces namespaces) {
479         if (sequence == null) {
480             return null;
481         }
482         final int c = sequence.charAt(0);
483         if (c == '_') {
484             return parseBNode(sequence);
485         } else if (c == '"' || c == '\'') {
486             return parseLiteral(sequence, namespaces);
487         } else {
488             return parseURI(sequence, namespaces);
489         }
490     }

492     private static URI parseURI(final CharSequence sequence, @Nullable final Namespaces namespaces) {
493 
494         if (sequence.charAt(0) == '<') {
495             final int last = sequence.length() - 1;
496             final StringBuilder builder = new StringBuilder(last - 1);
497             if (sequence.charAt(last) != '>') {
498                 throw new IllegalArgumentException("Invalid URI: " + sequence);
499             }
500             int i = 1;
501             while (i < last) {
502                 char c = sequence.charAt(i++);
503                 if (c < 32) { // discard control chars but accept other chars forbidden by W3C
504                     throw new IllegalArgumentException("Invalid char '" + c + "' in URI: "
505                             + sequence);
506                 } else if (c != '\\') {
507                     builder.append(c);
508                 } else {
509                     if (i == last) {
510                         throw new IllegalArgumentException("Invalid URI: " + sequence);
511                     }
512                     c = sequence.charAt(i++);
513                     if (c == 'u') {
514                         builder.append(parseHex(sequence, i, 4));
515                         i += 4;
516                     } else if (c == 'U') {
517                         builder.append(parseHex(sequence, i, 8));
518                         i += 8;
519                     } else {
520                         builder.append(c); // accept \> and \\ plus others
521                     }
522                 }
523             }
524             return VALUE_FACTORY.createURI(builder.toString());
525 
526         } else if (namespaces != null) {
527             final int len = sequence.length();
528             final StringBuilder builder = new StringBuilder(len);
529 
530             int i = 0;
531             while (i < len) {
532                 final char c = sequence.charAt(i++);
533                 if (c == ':') {
534                     if (i > 2 && !isPN_CHARS(sequence.charAt(i - 2))) {
535                         throw new IllegalArgumentException("Invalid qname " + sequence);
536                     }
537                     break;
538                 } else if (i == 1 && !isPN_CHARS_BASE(c) || i > 1 && !isPN_CHARS(c) && c != '.') {
539                     throw new IllegalArgumentException("Invalid qname " + sequence);
540                 } else {
541                     builder.append(c);
542                 }
543             }
544             final String prefix = builder.toString();
545             final String namespace = namespaces.uriFor(prefix);
546             if (namespace == null) {
547                 throw new IllegalArgumentException("Unknown prefix " + prefix);
548             }
549             builder.setLength(0);
550 
551             while (i < len) {
552                 final char c = sequence.charAt(i++);
553                 if (c == '%') {
554                     builder.append(parseHex(sequence, i, 2));
555                     i += 2;
556                 } else if (c == '\\') {
557                     final char d = sequence.charAt(i++);
558                     if (d == '_' || d == '~' || d == '.' || d == '-' || d == '!' || d == '$'
559                             || d == '&' || d == '\'' || d == '(' || d == ')' || d == '*'
560                             || d == '+' || d == ',' || d == ';' || d == '=' || d == '/'
561                             || d == '?' || d == '#' || d == '@' || d == '%') {
562                         builder.append(d);
563                     } else {
564                         throw new IllegalArgumentException("Invalid qname " + sequence);
565                     }
566                 } else if (isPN_CHARS_BASE(c) || c == ':' || i == 1 && (c == '_' || isNumber(c))
567                         || i > 1 && (isPN_CHARS(c) || i < len && c == '.')) {
568                     builder.append(c);
569                 } else {
570                     throw new IllegalArgumentException("Invalid qname " + sequence);
571                 }
572             }
573             final String name = builder.toString();
574 
575             return VALUE_FACTORY.createURI(namespace, name);
576         }
577 
578         throw new IllegalArgumentException("Unsupported qname " + sequence);
579     }

581     private static BNode parseBNode(final CharSequence sequence) {
582         final int len = sequence.length();
583         if (len > 2 && sequence.charAt(1) == ':') {
584             final StringBuilder builder = new StringBuilder(len - 2);
585             boolean ok = true;
586             char c = sequence.charAt(2);
587             builder.append(c);
588             ok &= isPN_CHARS_U(c) || isNumber(c);
589             for (int i = 2; i < len - 1; ++i) {
590                 c = sequence.charAt(i);
591                 builder.append(c);
592                 ok &= isPN_CHARS(c) || c == '.';
593             }
594             c = sequence.charAt(len - 1);
595             builder.append(c);
596             ok &= isPN_CHARS(c);
597             if (ok) {
598                 return VALUE_FACTORY.createBNode(builder.toString());
599             }
600         }
601         throw new IllegalArgumentException("Invalid BNode '" + sequence + "'");
602     }

604     private static Literal parseLiteral(final CharSequence sequence,
605             @Nullable final Namespaces namespaces) {
606 
607         final StringBuilder builder = new StringBuilder(sequence.length());
608         final int len = sequence.length();
609 
610         final char delim = sequence.charAt(0);
611         char c = 0;
612         int i = 1;
613         while (i < len && (c = sequence.charAt(i++)) != delim) {
614             if (c == '\\' && i < len) {
615                 c = sequence.charAt(i++);
616                 switch (c) {
617                 case 'b':
618                     builder.append('\b');
619                     break;
620                 case 'f':
621                     builder.append('\f');
622                     break;
623                 case 'n':
624                     builder.append('\n');
625                     break;
626                 case 'r':
627                     builder.append('\r');
628                     break;
629                 case 't':
630                     builder.append('\t');
631                     break;
632                 case 'u':
633                     builder.append(parseHex(sequence, i, 4));
634                     i += 4;
635                     break;
636                 case 'U':
637                     builder.append(parseHex(sequence, i, 8));
638                     i += 8;
639                     break;
640                 default:
641                     builder.append(c); // handles ' " \
642                     break;
643                 }
644             } else {
645                 builder.append(c);
646             }
647         }
648         final String label = builder.toString();
649 
650         if (i == len && c == delim) {
651             return VALUE_FACTORY.createLiteral(label);
652 
653         } else if (i < len - 2 && sequence.charAt(i) == '^' && sequence.charAt(i + 1) == '^') {
654             final URI datatype = parseURI(sequence.subSequence(i + 2, len), namespaces);
655             return VALUE_FACTORY.createLiteral(label, datatype);
656 
657         } else if (i < len - 1 && sequence.charAt(i) == '@') {
658             builder.setLength(0);
659             boolean minusFound = false;
660             for (int j = i + 1; j < len; ++j) {
661                 c = sequence.charAt(j);
662                 if (!isLetter(c) && (c != '-' || j == i + 1 || j == len - 1)
663                         && (!isNumber(c) || !minusFound)) {
664                     throw new IllegalArgumentException("Invalid lang in '" + sequence + "'");
665                 }
666                 minusFound |= c == '-';
667                 builder.append(c);
668             }
669             return VALUE_FACTORY.createLiteral(label, builder.toString());
670         }
671 
672         throw new IllegalArgumentException("Invalid literal '" + sequence + "'");
673     }

675     private static char parseHex(final CharSequence sequence, final int index, final int count) {
676         int code = 0;
677         final int len = sequence.length();
678         if (index + count >= len) {
679             throw new IllegalArgumentException("Incomplete hex code '"
680                     + sequence.subSequence(index, len) + "' in RDF value '" + sequence + "'");
681         }
682         for (int i = 0; i < count; ++i) {
683             final char c = sequence.charAt(index + i);
684             final int digit = Character.digit(c, 16);
685             if (digit < 0) {
686                 throw new IllegalArgumentException("Invalid hex digit '" + c + "' in RDF value '"
687                         + sequence + "'");
688             }
689             code = code * 16 + digit;
690         }
691         return (char) code;
692     }

1155             this.rankedNamespaces = Arrays.asList(rankedNamespaces);

1158         @Override
1159         public int compare(final Value v1, final Value v2) {
1160             if (v1 instanceof URI) {
1161                 if (v2 instanceof URI) {
1162                     final int rank1 = this.rankedNamespaces.indexOf(((URI) v1).getNamespace());
1163                     final int rank2 = this.rankedNamespaces.indexOf(((URI) v2).getNamespace());
1164                     if (rank1 >= 0 && (rank1 < rank2 || rank2 < 0)) {
1165                         return -1;
1166                     } else if (rank2 >= 0 && (rank2 < rank1 || rank1 < 0)) {
1167                         return 1;
1168                     }
1169                     final String string1 = Statements.formatValue(v1, Namespaces.DEFAULT);
1170                     final String string2 = Statements.formatValue(v2, Namespaces.DEFAULT);
1171                     return string1.compareTo(string2);
1172                 } else {
1173                     return -1;
1174                 }
1175             } else if (v1 instanceof BNode) {
1176                 if (v2 instanceof BNode) {
1177                     return ((BNode) v1).getID().compareTo(((BNode) v2).getID());
1178                 } else if (v2 instanceof URI) {
1179                     return 1;
1180                 } else {
1181                     return -1;
1182                 }
1183             } else if (v1 instanceof Literal) {
1184                 if (v2 instanceof Literal) {
1185                     return ((Literal) v1).getLabel().compareTo(((Literal) v2).getLabel());
1186                 } else if (v2 instanceof Resource) {
1187                     return 1;
1188                 } else {
1189                     return -1;
1190                 }
1191             } else {
1192                 if (v1 == v2) {
1193                     return 0;
1194                 } else {
1195                     return 1;
1196                 }
1197             }
1198         }

1210             this.components = components.trim().toLowerCase();
1211             this.valueComparator = Objects.requireNonNull(valueComparator);
1212             for (int i = 0; i < this.components.length(); ++i) {
1213                 final char c = this.components.charAt(i);
1214                 if (c != 's' && c != 'p' && c != 'o' && c != 'c') {
1215                     throw new IllegalArgumentException("Invalid components: " + components);
1216                 }
1217             }
1218         }
1219 
1220         @Override
1221         public int compare(final Statement s1, final Statement s2) {
1222             for (int i = 0; i < this.components.length(); ++i) {
1223                 final char c = this.components.charAt(i);
1224                 final Value v1 = getValue(s1, c);
1225                 final Value v2 = getValue(s2, c);
1226                 final int result = this.valueComparator.compare(v1, v2);
1227                 if (result != 0) {
1228                     return result;
1229                 }
1230             }
1231             return 0;
1232         }
1233 
1234         private Value getValue(final Statement statement, final char component) {
1235             switch (component) {
1236             case 's':
1237                 return statement.getSubject();
1238             case 'p':
1239                 return statement.getPredicate();
1240             case 'o':
1241                 return statement.getObject();
1242             case 'c':
1243                 return statement.getContext();
1244             default:
1245                 throw new Error();
1246             }
1247         }

1251     private static final class StatementMatcher implements Predicate<Statement> {
1252 
1253         @Nullable
1254         private final ValueMatcher subjMatcher;
1255 
1256         @Nullable
1257         private final ValueMatcher predMatcher;
1258 
1259         @Nullable
1260         private final ValueMatcher objMatcher;
1261 
1262         @Nullable
1263         private final ValueMatcher ctxMatcher;
1264 
1265         @SuppressWarnings("unchecked")
1266         public StatementMatcher(final String spec) {
1267 
1268             // Initialize the arrays used to create the ValueTransformers
1269             final List<?>[] expressions = new List<?>[4];
1270             final Boolean[] includes = new Boolean[4];
1271             for (int i = 0; i < 4; ++i) {
1272                 expressions[i] = new ArrayList<String>();
1273             }
1274 
1275             // Parse the specification string
1276             char action = 0;
1277             final List<Integer> components = new ArrayList<Integer>();
1278             for (final String token : spec.split("\\s+")) {
1279                 final char ch0 = token.charAt(0);
1280                 if (ch0 == '+' || ch0 == '-') {
1281                     action = ch0;
1282                     if (token.length() == 1) {
1283                         throw new IllegalArgumentException("No component(s) specified in '" + spec
1284                                 + "'");
1285                     }
1286                     components.clear();
1287                     for (int i = 1; i < token.length(); ++i) {
1288                         final char ch1 = Character.toLowerCase(token.charAt(i));
1289                         final int component = ch1 == 's' ? 0 : ch1 == 'p' ? 1 : ch1 == 'o' ? 2
1290                                 : ch1 == 'c' ? 3 : -1;
1291                         if (component < 0) {
1292                             throw new IllegalArgumentException("Invalid component '" + ch1
1293                                     + "' in '" + spec + "'");
1294                         }
1295                         components.add(component);
1296                     }
1297                 } else if (action == 0) {
1298                     throw new IllegalArgumentException("Missing selector in '" + spec + "'");
1299                 } else {
1300                     for (final int component : components) {
1301                         ((List<String>) expressions[component]).add(token);
1302                         final Boolean include = action == '+' ? Boolean.TRUE : Boolean.FALSE;
1303                         if (includes[component] != null
1304                                 && !Objects.equals(includes[component], include)) {
1305                             throw new IllegalArgumentException(
1306                                     "Include (+) and exclude (-) rules both "
1307                                             + "specified for same component in '" + spec + "'");
1308                         }
1309                         includes[component] = include;
1310                     }
1311                 }
1312             }
1313 
1314             // Create ValueTransformers
1315             final ValueMatcher[] matchers = new ValueMatcher[4];
1316             for (int i = 0; i < 4; ++i) {
1317                 matchers[i] = expressions[i].isEmpty() ? null : new ValueMatcher(
1318                         (List<String>) expressions[i], Boolean.TRUE.equals(includes[i]));
1319             }
1320             this.subjMatcher = matchers[0];
1321             this.predMatcher = matchers[1];
1322             this.objMatcher = matchers[2];
1323             this.ctxMatcher = matchers[3];
1324         }
1325 
1326         @Override
1327         public boolean test(final Statement stmt) {
1328             return (this.subjMatcher == null || this.subjMatcher.match(stmt.getSubject()))
1329                     && (this.predMatcher == null || this.predMatcher.match(stmt.getPredicate()))
1330                     && (this.objMatcher == null || this.objMatcher.match(stmt.getObject()))
1331                     && (this.ctxMatcher == null || this.ctxMatcher.match(stmt.getContext()));
1332         }

1334         private static final class ValueMatcher {
1335 
1336             private final boolean include;
1337 
1338             // for URIs
1339 
1340             private final boolean matchAnyURI;
1341 
1342             private final Set<String> matchedURINamespaces;
1343 
1344             private final Set<URI> matchedURIs;
1345 
1346             // for BNodes
1347 
1348             private final boolean matchAnyBNode;
1349 
1350             private final Set<BNode> matchedBNodes;
1351 
1352             // for Literals
1353 
1354             private final boolean matchAnyPlainLiteral;
1355 
1356             private final boolean matchAnyLangLiteral;
1357 
1358             private final boolean matchAnyTypedLiteral;
1359 
1360             private final Set<String> matchedLanguages;
1361 
1362             private final Set<URI> matchedDatatypeURIs;
1363 
1364             private final Set<String> matchedDatatypeNamespaces;
1365 
1366             private final Set<Literal> matchedLiterals;
1367 
1368             ValueMatcher(final Iterable<String> matchExpressions, final boolean include) {
1369 
1370                 this.include = include;
1371 
1372                 this.matchedURINamespaces = new HashSet<>();
1373                 this.matchedURIs = new HashSet<>();
1374                 this.matchedBNodes = new HashSet<>();
1375                 this.matchedLanguages = new HashSet<>();
1376                 this.matchedDatatypeURIs = new HashSet<>();
1377                 this.matchedDatatypeNamespaces = new HashSet<>();
1378                 this.matchedLiterals = new HashSet<>();
1379 
1380                 boolean matchAnyURI = false;
1381                 boolean matchAnyBNode = false;
1382                 boolean matchAnyPlainLiteral = false;
1383                 boolean matchAnyLangLiteral = false;
1384                 boolean matchAnyTypedLiteral = false;
1385 
1386                 for (final String expression : matchExpressions) {
1387                     if ("<*>".equals(expression)) {
1388                         matchAnyURI = true;
1389                     } else if ("_:*".equals(expression)) {
1390                         matchAnyBNode = true;
1391                     } else if ("*".equals(expression)) {
1392                         matchAnyPlainLiteral = true;
1393                     } else if ("*@*".equals(expression)) {
1394                         matchAnyLangLiteral = true;
1395                     } else if ("*^^*".equals(expression)) {
1396                         matchAnyTypedLiteral = true;
1397                     } else if (expression.startsWith("*@")) {
1398                         this.matchedLanguages.add(expression.substring(2));
1399                     } else if (expression.startsWith("*^^")) {
1400                         if (expression.endsWith(":*")) {
1401                             this.matchedDatatypeNamespaces.add(Namespaces.DEFAULT
1402                                     .uriFor(expression.substring(3, expression.length() - 2)));
1403                         } else {
1404                             this.matchedDatatypeURIs.add((URI) Statements.parseValue(
1405                                     expression.substring(3), Namespaces.DEFAULT));
1406                         }
1407                     } else if (expression.endsWith(":*")) {
1408                         this.matchedURINamespaces.add(Namespaces.DEFAULT.uriFor(expression
1409                                 .substring(0, expression.length() - 2)));
1410 
1411                     } else if (expression.endsWith("*>")) {
1412                         this.matchedURINamespaces.add(expression.substring(1,
1413                                 expression.length() - 2));
1414                     } else {
1415                         final Value value = Statements.parseValue(expression, Namespaces.DEFAULT);
1416                         if (value instanceof URI) {
1417                             this.matchedURIs.add((URI) value);
1418                         } else if (value instanceof BNode) {
1419                             this.matchedBNodes.add((BNode) value);
1420                         } else if (value instanceof Literal) {
1421                             this.matchedLiterals.add((Literal) value);
1422                         }
1423 
1424                     }
1425                 }
1426 
1427                 this.matchAnyURI = matchAnyURI;
1428                 this.matchAnyBNode = matchAnyBNode;
1429                 this.matchAnyPlainLiteral = matchAnyPlainLiteral;
1430                 this.matchAnyLangLiteral = matchAnyLangLiteral;
1431                 this.matchAnyTypedLiteral = matchAnyTypedLiteral;
1432             }
1433 
1434             boolean match(final Value value) {
1435                 final boolean matched = matchHelper(value);
1436                 return this.include == matched;
1437             }
1438 
1439             private boolean matchHelper(final Value value) {
1440                 if (value instanceof URI) {
1441                     return this.matchAnyURI //
1442                             || contains(this.matchedURIs, value)
1443                             || containsNs(this.matchedURINamespaces, (URI) value);
1444                 } else if (value instanceof Literal) {
1445                     final Literal lit = (Literal) value;
1446                     final String lang = lit.getLanguage();
1447                     final URI dt = lit.getDatatype();
1448                     return lang == null
1449                             && (dt == null || XMLSchema.STRING.equals(dt))
1450                             && this.matchAnyPlainLiteral //
1451                             || lang != null //
1452                             && (this.matchAnyLangLiteral || contains(this.matchedLanguages, lang)) //
1453                             || dt != null //
1454                             && (this.matchAnyTypedLiteral
1455                                     || contains(this.matchedDatatypeURIs, dt) || containsNs(
1456                                         this.matchedDatatypeNamespaces, dt)) //
1457                             || contains(this.matchedLiterals, lit);
1458                 } else {
1459                     return this.matchAnyBNode //
1460                             || contains(this.matchedBNodes, value);
1461                 }
1462             }
1463 
1464             private static boolean contains(final Set<?> set, final Object value) {
1465                 return !set.isEmpty() && set.contains(value);
1466             }
1467 
1468             private static boolean containsNs(final Set<String> set, final URI uri) {
1469                 if (set.isEmpty()) {
1470                     return false;
1471                 }
1472                 if (set.contains(uri.getNamespace())) {
1473                     return true; // exact lookup
1474                 }
1475                 final String uriString = uri.stringValue();
1476                 for (final String elem : set) {
1477                     if (uriString.startsWith(elem)) {
1478                         return true; // prefix match
1479                     }
1480                 }
1481                 return false;
1482             }
1483 
1484         }
