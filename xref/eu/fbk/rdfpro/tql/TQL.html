14  package eu.fbk.rdfpro.tql;
15  
16  import java.nio.charset.Charset;
17  
18  import org.openrdf.rio.RDFFormat;
19  
20  /**
21   * Constants for the Turtle Quads (TQL) format.
22   * <p>
23   * The Turtle Quads {@link RDFFormat} is defined by constant {@link #FORMAT}. As this constant is
24   * not part of the predefined set of formats in {@link RDFFormat}, it is necessary to register it.
25   * This can be done either via {@link RDFFormat#register(RDFFormat)}, or by simply calling method
26   * {@link #register()} on this class, which ensures that multiple calls will result in a single
27   * registration.
28   * </p>
29   */
30  public final class TQL {
31  
32      /** RDFFormat constant for the Turtle Quads (TQL) format). */
33      public static final RDFFormat FORMAT = new RDFFormat("Turtle Quads", "application/x-tql",
34              Charset.forName("UTF-8"), "tql", false, true);
35  
36      static {
37          RDFFormat.register(FORMAT);
38      }
39  
40      /**
41       * Registers the Turtle Quads format in the RIO registry. Calling this method multiple times
42       * results in a single registration. Note that registration is also done transparently the
43       * first time this class is accessed.
44       */
45      public static void register() {
46          // Calling this method will cause the static initializer to run once
47      }
48  
49      // Package protected utility methods
50  
51      static boolean isPN_CHARS(final int c) { // ok
52          return isPN_CHARS_U(c) || isNumber(c) || c == '-' || c == 0x00B7 || c >= 0x0300
53                  && c <= 0x036F || c >= 0x203F && c <= 0x2040;
54      }
55  
56      static boolean isPN_CHARS_U(final int c) { // ok
57          return isPN_CHARS_BASE(c) || c == '_';
58      }
59  
60      static boolean isPN_CHARS_BASE(final int c) { // ok
61          return isLetter(c) || c >= 0x00C0 && c <= 0x00D6 || c >= 0x00D8 && c <= 0x00F6
62                  || c >= 0x00F8 && c <= 0x02FF || c >= 0x0370 && c <= 0x037D || c >= 0x037F
63                  && c <= 0x1FFF || c >= 0x200C && c <= 0x200D || c >= 0x2070 && c <= 0x218F
64                  || c >= 0x2C00 && c <= 0x2FEF || c >= 0x3001 && c <= 0xD7FF || c >= 0xF900
65                  && c <= 0xFDCF || c >= 0xFDF0 && c <= 0xFFFD || c >= 0x10000 && c <= 0xEFFFF;
66      }
67  
68      static boolean isLetter(final int c) {
69          return c >= 65 && c <= 90 || c >= 97 && c <= 122;
70      }
71  
72      static boolean isNumber(final int c) {
73          return c >= 48 && c <= 57;
74      }
75  
76  }
