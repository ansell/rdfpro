31  /**
32   * Test case for {@link eu.fbk.rdfpro.base.FilterProcessorOld}.
33   *
34   * @author Michele Mostarda (mostarda@fbk.eu)
35   */
36  public class TransformProcessorTest {
37  
38      @Test
39      public void testFilter() throws RDFHandlerException {
40          // Create a filter which matches any subject and object with predicates of statements
41          // contained into the file.nt.
42          final RDFProcessor filter = RDFProcessors.parse(true, "@filter 'sou +[./file.nt]p -*'");
43          Assert.assertEquals(0, filter.getExtraPasses());
44          final int[] flags = new int[2];
45          final List<Statement> statements = new ArrayList<>();
46          final RDFHandler handler = filter.wrap(new RDFHandler() {
47  
48              @Override
49              public void startRDF() throws RDFHandlerException {
50                  flags[0]++;
51              }
52  
53              @Override
54              public void endRDF() throws RDFHandlerException {
55                  flags[1]++;
56              }
57  
58              @Override
59              public void handleNamespace(final String pref, final String name)
60                      throws RDFHandlerException {
61                  throw new IllegalStateException();
62              }
63  
64              @Override
65              public void handleStatement(final Statement s) throws RDFHandlerException {
66                  statements.add(s);
67              }
68  
69              @Override
70              public void handleComment(final String comment) throws RDFHandlerException {
71                  throw new IllegalStateException();
72              }
73          });
74  
75          final ValueFactory factory = new ValueFactoryImpl();
76          handler.startRDF();
77          handler.handleStatement(factory.createStatement(factory.createURI("http://sx"),
78                  factory.createURI("http://p1"), factory.createURI("http://sx")));
79          handler.handleStatement(factory.createStatement(factory.createURI("http://sF"),
80                  factory.createURI("http://pF"), factory.createURI("http://sF")));
81          handler.handleStatement(factory.createStatement(factory.createURI("http://sy"),
82                  factory.createURI("http://p2"), factory.createURI("http://sy")));
83          handler.endRDF();
84  
85          Assert.assertEquals(1, flags[0]);
86          Assert.assertEquals(1, flags[1]);
87          Assert.assertEquals(2, statements.size());
88          Assert.assertEquals("(http://sx, http://p1, http://sx)", statements.get(0).toString());
89          Assert.assertEquals("(http://sy, http://p2, http://sy)", statements.get(1).toString());
90      }
